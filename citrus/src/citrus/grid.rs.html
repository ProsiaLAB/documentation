127pub fn pre_define(par: &mut Parameters, gp: &mut Vec<Grid>) -> Result<()> {
128    let mut rand_gen = if true {
129        // Use fixed seed for reproducibility
130        // Note: SeedableRng::seed_from_u64 takes a u64 seed
131        StdRng::seed_from_u64(6611304)
132    } else {
133        // Seed from the system's entropy source for non-reproducible randomness
134        // StdRng::from_entropy is a good way to get a random seed
135        StdRng::try_from_os_rng().expect("Failed to seed random number generator from entropy")
136    };
137    par.num_densities = 1;
138    for dens in gp.iter_mut().map(|g| &mut g.dens) {
139        dens.fill(0.0);
140    }
141
142    // Open grid file
143    let file = File::open(&par.pre_grid)?;
144    let reader = BufReader::new(file);
145
146    for (i, line) in reader.lines().enumerate().take(par.p_intensity) {
147        let line = line?;
148        let values: RVector = line
149            .split_whitespace()
150            .map(|v| {
151                v.parse::<f64>().map_err(|e| {
152                    eprintln!("Failed to parse {}: {}", v, e);
153                    e
154                })
155            })
156            .collect::<Result<RVector, ParseFloatError>>()?;
157
158        if values.len() != 9 {
159            bail!("Expected 9 values");
160        }
161
162        let id = values[0] as i32;
163        if id >= par.p_intensity as i32 {
164            bail!("Invalid grid point ID: {}", id);
165        }
166
167        gp[i].id = id;
168        gp[i].x[0] = values[1];
169        gp[i].x[1] = values[2];
170        gp[i].x[2] = values[3];
171
172        gp[i].dens[0] = values[4];
173
174        gp[i].t[0] = values[5];
175        gp[i].t[1] = values[5];
176
177        gp[i].vel[0] = values[6];
178        gp[i].vel[1] = values[7];
179        gp[i].vel[2] = values[8];
180
181        gp[i].dopb_turb = 200.0;
182
183        if let Some(ref mut molecule) = gp[i].mol {
184            molecule[0].abun = 1e-9;
185        } else {
186            bail!("No molecular data found");
187        }
188
189        gp[i].sink = false;
190        gp[i].dir = Vec::new();
191        gp[i].neigh = Vec::new();
192        gp[i].mag_field = RVector::zeros(3);
193
194        utils::progress_bar(i as f64 / par.p_intensity as f64, 50);
195    }
196
197    // check grid densities
198    if par.do_mol_calcs {
199        check_grid_densities(gp, par);
200    }
201
202    // if random generator was set assign densities for the rest of the grid
203    let mut i = par.p_intensity; // Initialize `i` manually since we want to control it
204    while i < par.ncell {
205        let x = rand_gen.random_range(-1.0..1.0);
206        let y = rand_gen.random_range(-1.0..1.0);
207        let z = rand_gen.random_range(-1.0..1.0);
208
209        if (x * x + y * y + z * z) < 1.0 {
210            let scale = par.radius * (1.0f64 / (x * x + y * y + z * z)).sqrt();
211
212            gp[i].id = i as i32;
213            gp[i].x[0] = x * scale;
214            gp[i].x[1] = y * scale;
215            gp[i].x[2] = z * scale;
216            gp[i].sink = true;
217
218            // Update the molecule data if it's available
219            if let Some(molecule) = &mut gp[i].mol {
220                molecule[0].abun = 0.0;
221                molecule[0].nmol = 0.0;
222            } else {
223                bail!("No molecular data found");
224            }
225
226            gp[i].dens[0] = cc::CITRUS_GLOBAL_EPS; // Assuming CITRUS_EPS is defined
227            gp[i].t[0] = par.cmb_temp;
228            gp[i].t[1] = par.cmb_temp;
229            gp[i].mag_field = RVector::zeros(3);
230            gp[i].dopb_turb = 0.0;
231
232            // Initialize velocity array to 0.0
233            for j in 0..N_DIMS {
234                gp[i].vel[j] = 0.0;
235            }
236
237            // Continue to next iteration
238            i += 1;
239        } else {
240            // If the condition is not met, retry this iteration
241            // Do not increment `i` when the condition fails
242        }
243    }
244
245    // call Delaunay triangulation
246    let _ = delaunay(gp, par.ncell, false, true)?;
247
248    /* We just asked delaunay() to flag any grid points with IDs lower than
249     * par->pIntensity (which means their distances from model centre are less
250     * than the model radius) but which are nevertheless found to be sink points
251     * by virtue of the geometry of the mesh of Delaunay cells. Now we need to
252     * reshuffle the list of grid points, then reset par->pIntensity, such that
253     * all the non-sink points still have IDs lower than par->pIntensity.
254     */
255
256    let n_extra_sinks = reorder_grid(gp, par.ncell)?;
257    par.p_intensity -= n_extra_sinks as usize;
258    par.p_intensity += n_extra_sinks as usize;
259
260    dist_calc(gp, par.ncell);
261
262    if !par.grid_file.is_empty() {
263        write_vtk_unstructured_points(gp, par)?;
264    }
265
266    Ok(())
267}

331/// The principal purpose of this function is to perform a Delaunay triangulation
332/// for the set of points defined by the input argument 'g'. This is achieved via
333/// routines in the 3rd-party package qhull.
334///
335/// A note about qhull nomenclature: a vertex is what you think it is - i.e., a
336/// point; but a facet means in this context a Delaunay triangle (in 2D) or
337/// tetrahedron (in 3D). This nomenclature arises because the Delaunay cells are
338/// indeed facets (or rather projections of facets) of the convex hull constructed
339/// in 1 higher dimension.
340///
341/// Required elements of structs:
342///         struct grid *gp:
343///                 .id
344///                 .x
345///
346/// Elements of structs are set as follows:
347///         struct grid *gp:
348///                 .sink
349///                 .numNeigh
350///                 .neigh (this is malloc'd too large and at present not
351/// realloc'd.)
352///
353///         cellType *dc (if getCells>0):
354///                 .id
355///                 .neigh
356///                 .vertx
357pub fn delaunay(
358    gp: &mut [Grid],
359    num_points: usize,
360    get_cells: bool,
361    check_sink: bool,
362) -> Result<DelaunayResult> {
363    // pt_array contains the grid point locations in the format required by qhull.
364    let mut pt_array = Vec::with_capacity(num_points * N_DIMS);
365    for gpi in gp.iter().take(num_points) {
366        for di in 0..N_DIMS {
367            pt_array.push(gpi.x[di]);
368        }
369    }
370
371    let qh = QhBuilder::default()
372        .delaunay(true)
373        .scale_last(true)
374        .triangulate(true)
375        .build_managed(N_DIMS, pt_array)
376        .map_err(|e| anyhow!("Failed to build Qhull: {:?}", e))?;
377
378    let mut indices: Vec<usize> = Vec::new();
379
380    if check_sink {
381        for facet in qh.all_facets() {
382            if !facet.upper_delaunay() {
383                let neighbors = facet
384                    .neighbors()
385                    .ok_or_else(|| anyhow!("Failed to get neighbors"))?;
386                for neighbor in neighbors.iter() {
387                    if neighbor.upper_delaunay() {
388                        let vertices = neighbor
389                            .vertices()
390                            .ok_or_else(|| anyhow!("Failed to get vertices"))?;
391                        for vertex in vertices.iter() {
392                            let ppi = vertex.point_id(&qh);
393                            match ppi {
394                                Ok(ppi) => {
395                                    indices.push(ppi as usize);
396                                }
397                                Err(_) => {
398                                    bail!("Failed to get point ID");
399                                }
400                            }
401                        }
402                    }
403                }
404            }
405        }
406    }
407
408    for &index in indices.iter() {
409        gp[index].sink = true;
410    }
411
412    for vertex in qh.vertices() {
413        let id = vertex.point_id(&qh);
414        match id {
415            Ok(id) => {
416                let neighbors = vertex
417                    .neighbors()
418                    .ok_or_else(|| anyhow!("Failed to get neighbors"))?;
419                gp[id as usize].num_neigh = neighbors.size(&qh);
420                if gp[id as usize].num_neigh == 0 {
421                    bail!("`qhull` failed silently. A smoother `grid_density` might help.");
422                }
423                gp[id as usize].neigh = {
424                    let mut v = Vec::with_capacity(gp[id as usize].num_neigh);
425                    for _ in 0..gp[id as usize].num_neigh {
426                        v.push(None);
427                    }
428                    v
429                };
430            }
431            Err(_) => {
432                bail!("Failed to get point ID");
433            }
434        }
435    }
436
437    let mut point_ids_this_facet: [usize; N_DIMS + 1] = [0; N_DIMS + 1];
438    let mut num_cells: usize = 0;
439    for facet in qh.all_facets() {
440        if !facet.upper_delaunay() {
441            let mut j: usize = 0;
442            let vertices = facet
443                .vertices()
444                .ok_or_else(|| anyhow!("Failed to get vertices"))?;
445            for vertex in vertices.iter() {
446                let point_id_this = vertex.point_id(&qh);
447                match point_id_this {
448                    Ok(point_id_this) => {
449                        point_ids_this_facet[j] = point_id_this as usize;
450                        j += 1;
451                    }
452                    Err(_) => {
453                        bail!("Failed to get point ID");
454                    }
455                }
456            }
457            for i in 0..N_DIMS + 1 {
458                let id_i = point_ids_this_facet[i];
459                for (j, &id_j) in point_ids_this_facet.iter().enumerate().take(N_DIMS + 1) {
460                    if i != j {
461                        let mut k: usize = 0;
462                        while gp[id_i].neigh[k].is_some() {
463                            match gp[id_i].neigh[k] {
464                                Some(ref neigh_grid) => {
465                                    let grid_id_j = gp[id_j].id;
466                                    if neigh_grid.id != grid_id_j {
467                                        k += 1;
468                                    } else {
469                                        break;
470                                    }
471                                }
472                                None => {
473                                    bail!(
474                                        "`qhull` failed silently. A smoother `grid_density` might help."
475                                    );
476                                }
477                            }
478                        }
479                    }
480                }
481            }
482            num_cells += 1;
483        }
484    }
485
486    for gp_point in &mut gp[..num_points] {
487        gp_point.num_neigh = gp_point
488            .neigh
489            .iter()
490            .filter(|neigh| neigh.is_some())
491            .count();
492    }
493
494    if get_cells {
495        let mut cells: Vec<Cell> = Vec::with_capacity(num_cells);
496        let mut fi: usize = 0;
497        for facet in qh.all_facets() {
498            if !facet.upper_delaunay() {
499                cells[fi].id = facet.id() as usize;
500                fi += 1;
501            }
502        }
503        let mut fi: usize = 0;
504        for facet in qh.all_facets() {
505            if !facet.upper_delaunay() {
506                let neighbors = facet
507                    .neighbors()
508                    .ok_or_else(|| anyhow!("Failed to get neighbors"))?;
509                for (i, neighbor) in neighbors.iter().enumerate() {
510                    if neighbor.upper_delaunay() {
511                        cells[fi].neigh[i] = None;
512                    } else {
513                        let mut ffi: usize = 0;
514                        let mut neighbor_not_found = true;
515                        while ffi < num_cells && neighbor_not_found {
516                            if cells[ffi].id == neighbor.id() as usize {
517                                cells[fi].neigh[i] = Some(ffi);
518                                neighbor_not_found = false;
519                            }
520                            ffi += 1;
521                        }
522                        if ffi >= num_cells && neighbor_not_found {
523                            bail!("Something went wrong with the Delaunay triangulation");
524                        }
525                    }
526                }
527                fi += 1;
528            }
529        }
530        Ok(DelaunayResult::Cells(cells))
531    } else {
532        Ok(DelaunayResult::NoCells)
533    }
534}

536/// The algorithm works its way up the list of points with one index and down with
537/// another. The 'up' travel stops at the 1st sink point it finds, the 'down' at the
538/// 1st non-sink point. If at that point the 'up' index is lower in value than the
539/// 'down', the points are swapped. This is just a tiny bit tricky because we also
540/// need to make sure all the neigh pointers are swapped. That's why we make an
541/// ordered list of indices and perform the swaps on that as well.
542fn reorder_grid(gp: &mut Vec<Grid>, num_points: usize) -> Result<i32> {
543    let mut n_extra_sinks = 0;
544    let mut indices: Vec<usize> = vec![0; num_points];
545
546    for (i, index) in indices.iter_mut().enumerate().take(num_points) {
547        *index = i;
548    }
549
550    let mut up_i = 0;
551    let mut down_i = num_points - 1;
552    loop {
553        while up_i < num_points && !gp[up_i].sink {
554            up_i += 1;
555        }
556        while down_i > 0 && gp[down_i].sink {
557            down_i -= 1;
558        }
559        if up_i >= down_i {
560            break;
561        }
562        n_extra_sinks += 1;
563
564        indices.swap(up_i, down_i);
565
566        // do the swap
567        let (before, after) = gp.as_mut_slice().split_at_mut(down_i + 1);
568        mem::swap(&mut before[down_i], &mut after[up_i - down_i]);
569
570        // retain the id values as sequential
571        gp[down_i].id = down_i as i32;
572        gp[up_i].id = up_i as i32;
573    }
574
575    /*
576    Now we sort out the .neigh values. An example of how this should work is as
577    follows. Suppose we swapped points 30 and 41. We have fixed up the .id values,
578    but the swap is still shown in the 'indices' array. Thus we will have
579
580            gp[30].id == 30 (but all the other data is from 41)
581            gp[41].id == 41 (but all the other data is from 30)
582            indices[30] == 41
583            indices[41] == 30
584
585    Suppose further that the old value of gp[i].neigh[j] is &gp[30]. We detect that
586    we need to fix it (change it to &gp[41]) because ngi=&gp[30].id=30 !=
587    indices[ngi=30]=41. gp[i].neigh[j] is then reset to &gp[indices[30]] = &gp[41],
588    i.e. to point to the same data as used to be in location 30.
589      */
590
591    for i in 0..num_points {
592        for j in 0..gp[i].num_neigh {
593            if let Some(ref mut neigh_grid) = gp[i].neigh[j] {
594                let ng_i = neigh_grid.id;
595                if ng_i != indices[ng_i as usize] as i32 {
596                    let index = indices[ng_i as usize] as usize;
597                    let new_neigh = mem::take(&mut gp[index]);
598                    gp[i].neigh[j] = Some(Box::new(new_neigh));
599                }
600            }
601        }
602    }
603
604    Ok(n_extra_sinks)
605}

607/// Calculate the distance between grid points
608/// The distance between two points is calculated as the Euclidean distance
609/// between the two points.
610fn dist_calc(gp: &mut [Grid], num_points: usize) {
611    for gp_point in &mut gp[..num_points] {
612        gp_point.dir = {
613            let mut v = Vec::with_capacity(gp_point.num_neigh);
614            for _ in 0..gp_point.num_neigh {
615                v.push(Point::default());
616            }
617            v
618        };
619        gp_point.ds.fill(0.0);
620
621        for (k, (dir, ds)) in gp_point
622            .dir
623            .iter_mut()
624            .zip(gp_point.ds.iter_mut())
625            .enumerate()
626        {
627            if let Some(ref neigh) = gp_point.neigh[k] {
628                dir.x
629                    .iter_mut()
630                    .zip(neigh.x.iter().zip(&gp_point.x))
631                    .for_each(|(dir_x, (&neigh_x, &gp_x))| *dir_x = neigh_x - gp_x);
632
633                *ds = dir.x.iter().map(|&v| v * v).sum::<f64>().sqrt();
634
635                dir.xn
636                    .iter_mut()
637                    .zip(&dir.x)
638                    .for_each(|(xn, &x)| *xn = x / *ds);
639            }
640        }
641
642        // gp_point.nphot = defaults::RAYS_PER_POINT;
643    }
644}

664    writeln!(file, "POINTS {} double", par.ncell)?;
665
666    for gp_i in &gp[..par.ncell] {
667        writeln!(
668            file,
669            "{:.6e} {:.6e} {:.6e}",
670            gp_i.x[0], gp_i.x[1], gp_i.x[2]
671        )?;
672    }
673
674    let qh = QhBuilder::default()
675        .delaunay(true)
676        .scale_last(true)
677        .is_tracing(0)
678        .build_managed(N_DIMS, pt_array)
679        .map_err(|e| anyhow!("Failed to build Qhull: {:?}", e))?;
680
681    let mut l: usize = 0;
682
683    for facet in qh.all_facets() {
684        if !facet.upper_delaunay() {
685            l += 1;
686        }
687    }
688
689    writeln!(file, "CELLS {} {}", l, 5 * l)?;
690    for facet in qh.all_facets() {
691        if !facet.upper_delaunay() {
692            writeln!(file, "4 ")?;
693            let vertices = facet
694                .vertices()
695                .ok_or_else(|| anyhow!("Failed to get vertices"))?;
696            for vertex in vertices.iter() {
697                let point_id = vertex.point_id(&qh);
698                match point_id {
699                    Ok(point_id) => {
700                        writeln!(file, "{} ", point_id)?;
701                    }
702                    Err(_) => {
703                        bail!("Failed to get point ID");
704                    }
705                }
706            }
707        }
708    }
709
710    writeln!(file, "CELL_TYPES {}", l)?;
711    for _ in 0..l {
712        writeln!(file, "10")?;
713    }
714    writeln!(file, "POINT_DATA {}", par.ncell)?;
715    writeln!(file, "SCALARS H2_density float 1")?;
716    writeln!(file, "LOOKUP_TABLE default")?;
717    for gp_i in &gp[..par.ncell] {
718        writeln!(file, "{:.6e}", gp_i.dens[0])?;
719    }
720    writeln!(file, "SCALARS Mol_density float 1")?;
721    writeln!(file, "LOOKUP_TABLE default")?;
722    if par.n_species > 0 {
723        for gp_i in &gp[..par.ncell] {
724            let mol_pop = gp_i
725                .mol
726                .as_ref()
727                .ok_or_else(|| anyhow!("Failed to get molecular data"))?;
728            writeln!(file, "{:.6e}", mol_pop[0].abun * gp_i.dens[0])?;
729        }
730    } else {
731        for _ in 0..par.ncell {
732            writeln!(file, "{:.6e}", 0.0)?;
733        }
734    }
735    writeln!(file, "SCALARS Gas_temperature float 1")?;
736    writeln!(file, "LOOKUP_TABLE default")?;
737    for gp_i in &gp[..par.ncell] {
738        writeln!(file, "{:.6e}", gp_i.t[0])?;
739    }
740    writeln!(file, "SCALARS velocity float 1")?;
741    for gp_i in &gp[..par.ncell] {
742        let length = gp_i.vel.iter().map(|&v| v * v).sum::<f64>().sqrt();
743        if length > 0.0 {
744            writeln!(
745                file,
746                "{:.6e} {:.6e} {:.6e}",
747                gp_i.vel[0] / length,
748                gp_i.vel[1] / length,
749                gp_i.vel[2] / length
750            )?;
751        } else {
752            writeln!(
753                file,
754                "{:.6e} {:.6e} {:.6e}",
755                gp_i.vel[0], gp_i.vel[1], gp_i.vel[2]
756            )?;
757        }
758    }
759
760    Ok(())
761}
