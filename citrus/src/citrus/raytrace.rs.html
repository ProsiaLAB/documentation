88    cell_visited: Vec<bool>,
89    isimplex: usize,
90    entry_face_index: usize,
91    ncells_in_chain: usize,
92    len_chain_ptrs: usize,
93    chain_of_cell_ids: Vec<usize>,
94    cell_exit_intersects: Vec<Intersect>,
95}

97impl ChainContext {
98    fn build_ray_cell_chain(
99        &mut self,
100        x: &RVector,
101        dir: &RVector,
102        rtp: &RTPreparation,
103        simplices: &[Simplex],
104    ) -> Result<i32, RTCError> {
105        const NUM_FACES: usize = N_DIMS + 1;
106
107        let buffer_size = 1024;
108        let mut intersects = Vec::with_capacity(NUM_FACES);
109        let mut good_exit_fis = Vec::new();
110        let mut marginal_exit_fis = Vec::new();
111
112        let mut following_single_chain = true;
113        while following_single_chain {
114            self.cell_visited[self.isimplex] = true;
115            if self.ncells_in_chain >= self.len_chain_ptrs {
116                self.len_chain_ptrs += buffer_size;
117                self.chain_of_cell_ids.resize(self.len_chain_ptrs, 0);
118                self.cell_exit_intersects
119                    .resize_with(self.len_chain_ptrs, Intersect::default);
120            }
121            (self.chain_of_cell_ids)[self.ncells_in_chain] = self.isimplex;
122            for fi in 0..NUM_FACES {
123                if fi != self.entry_face_index
124                    && (simplices[self.isimplex].neigh[fi].is_none()
125                        || simplices[self.isimplex].neigh[fi]
126                            .is_some_and(|n| !(self.cell_visited)[n]))
127                {
128                    let face = rtp.simplices[self.isimplex].extract_face(fi, &rtp.vertex_coords);
129                    let mut intersect = face.intersect_line_with_face(x, dir)?;
130                    intersect.fi = fi;
131                    if intersect.orientation == Orientation::ExitFace {
132                        if intersect.coll_par - cc::CITRUS_RT_EPS > 0.0 {
133                            good_exit_fis.push(fi);
134                        } else if intersect.coll_par + cc::CITRUS_RT_EPS > 0.0 {
135                            marginal_exit_fis.push(fi);
136                        }
137                    }
138                    intersects.push(intersect);
139                }
140            }
141            if good_exit_fis.len() > 1 {
142                return Err(RTCError::MultipleCandidates);
143            } else if good_exit_fis.len() == 1 || marginal_exit_fis.len() == 1 {
144                let exit_fi = if good_exit_fis.len() == 1 {
145                    good_exit_fis[0]
146                } else {
147                    marginal_exit_fis[0]
148                };
149                (self.cell_exit_intersects)[self.ncells_in_chain] =
150                    mem::take(&mut intersects[exit_fi]);
151                self.ncells_in_chain += 1;
152                if simplices[self.isimplex].neigh[exit_fi].is_none() {
153                    self.chain_of_cell_ids.resize(self.ncells_in_chain, 0);
154                    self.cell_exit_intersects
155                        .resize_with(self.ncells_in_chain, Intersect::default);
156                    self.len_chain_ptrs = self.ncells_in_chain;
157                    return Ok(0);
158                } else if let Some(neigh_idx) = simplices[self.isimplex].neigh[exit_fi] {
159                    self.entry_face_index =
160                        simplices[neigh_idx].get_new_entry_face_index(self.isimplex)?;
161                    self.isimplex = neigh_idx;
162                } else {
163                    return Err(RTCError::NotFound);
164                };
165            } else {
166                following_single_chain = false;
167            }
168        }
169
170        if marginal_exit_fis.is_empty() {
171            return Ok(3);
172        }
173
174        let mut i = 0;
175        let mut status = 4;
176        while i < marginal_exit_fis.len() && status > 0 {
177            let exit_fi = marginal_exit_fis[i];
178            (self.cell_exit_intersects)[self.ncells_in_chain] = mem::take(&mut intersects[exit_fi]);
179            if simplices[self.isimplex].neigh[exit_fi].is_none() {
180                self.chain_of_cell_ids.resize(self.ncells_in_chain, 0);
181                self.cell_exit_intersects
182                    .resize_with(self.ncells_in_chain, Intersect::default);
183                self.len_chain_ptrs = self.ncells_in_chain;
184                return Ok(0);
185            } else if let Some(neigh_idx) = simplices[self.isimplex].neigh[exit_fi] {
186                self.entry_face_index =
187                    simplices[neigh_idx].get_new_entry_face_index(self.isimplex)?;
188                self.isimplex = neigh_idx;
189                status = self.build_ray_cell_chain(x, dir, rtp, simplices)?;
190            } else {
191                return Err(RTCError::NotFound);
192            };
193
194            i += 1;
195        }
196
197        Ok(status)
198    }
199}

203#[derive(Debug, Default)]
204pub struct Simplex {
205    pub id: usize,
206    pub vertex: UVector,
207    pub centres: RVector,
208    pub neigh: Vec<Option<usize>>, // use index, not reference
209}

211impl Simplex {
212    /// Given a simplex dc and the face index (in the range {0...numDims}) fi,
213    /// this returns the desired information about that face. Note that the ordering of
214    /// the elements of face.r[] is the same as the ordering of the vertices of the
215    /// simplex, dc[].vertx[]; just the vertex fi is omitted.
216    ///
217    /// Note that the element 'centre' of the faceType struct is mean to contain the
218    /// spatial coordinates of the centre of the simplex, not of the face. This is
219    /// designed to facilitate orientation of the face and thus to help determine
220    /// whether rays which cross it are entering or exiting the simplex.
221    fn extract_face(&self, fi: usize, vertex_coords: &RVector) -> Face {
222        let num_faces = N_DIMS + 1;
223        let mut face = Face::default();
224        let mut vvi = 0;
225        for vi in 0..num_faces {
226            if vi != fi {
227                let gi = self.vertex[vi];
228                for di in 0..N_DIMS {
229                    face.r[vvi][di] = vertex_coords[N_DIMS * gi + di];
230                }
231                vvi += 1;
232            }
233        }
234        for di in 0..N_DIMS {
235            face.simplex_centres[di] = self.centres[di];
236        }
237        face
238    }
239
240    /// Finds the index of the old cell in the face list of the new cell.
241    fn get_new_entry_face_index(&self, dci: usize) -> Result<usize, RTCError> {
242        let num_faces = N_DIMS + 1;
243        self.neigh
244            .iter()
245            .take(num_faces)
246            .enumerate()
247            .find_map(|(i, &neigh_idx)| {
248                if neigh_idx == Some(dci) {
249                    Some(i)
250                } else {
251                    None
252                }
253            })
254            .ok_or(RTCError::NotFound)
255    }
256}

258/// This struct is meant to record all relevant information about the
259/// intersection between a ray (defined by a direction unit vector 'dir' and a
260/// starting position 'r') and a face of a simplex.

262pub struct Intersect {

272    /// `coll_par` is a measure of how close to any edge of the face `r_int`

285#[derive(Debug, Default)]
286pub struct Face {
287    /// `r` is a list of the the `N` vertices of the face, each of which has `N`
288    /// cartesian components.
289    pub r: Vec<RVector>,
290    /// `simplex_centres` is a convenience pointer array which gives
291    /// the location of the geometric centres of the simplexes.
292    pub simplex_centres: RVector,
293}
294
295impl Face {
296    /// Each of the faces of a polytope in N spatial dimensions is itself a polytope in
297    /// N-1 dimensions. Each face can therefore be represented via the coordinates of
298    /// its vertices expressed in an (N-1)-dimensional frame oriented so as to be
299    /// parallel to the face. The function of the present routine is to perform this
300    /// decomposition and to return both the (N-1)-dimensional frame (specified via the
301    /// coordinates in N dimensions of its N-1 basis vectors) and the coordinates in it
302    /// of each of the M face vertices. (If the face is simplicial, M==N.)
303    ///
304    /// The calling routine should call freeFacePlusBasis() after it is finished with
305    /// the returned object.
306    ///
307    /// Note that numVertices (a.k.a. M) is expected to be >= numDims (a.k.a. N). The
308    /// thing would not make much sense otherwise.
309    fn basis(&self, nvertices: usize) -> FaceBasis {
310        let mut vs = vec![RVector::zeros(nvertices); nvertices - 1];
311        let mut facebasis = FaceBasis::new();
312        for di in 0..N_DIMS {
313            facebasis.origin[di] = self.r[0][di];
314        }
315        for (vi, vsi) in vs.iter_mut().enumerate().take(nvertices - 1) {
316            for di in 0..N_DIMS {
317                vsi[di] = self.r[vi + 1][di] - self.r[0][di];
318            }
319        }
320        for (i, _) in vs.iter().enumerate().take(N_DIMS - 1) {
321            for di in 0..N_DIMS {
322                facebasis.axes[i][di] = vs[i][di];
323            }
324            for j in 0..i {
325                let mut dotval = 0.0;
326                for di in 0..N_DIMS {
327                    dotval += facebasis.axes[i][di] * facebasis.axes[j][di];
328                }
329                for di in 0..N_DIMS {
330                    facebasis.axes[i][di] -= dotval * facebasis.axes[j][di];
331                }
332            }
333            let mut norm = 0.0;
334            for di in 0..N_DIMS {
335                norm += facebasis.axes[i][di] * facebasis.axes[i][di];
336            }
337            norm = 1.0 / norm.sqrt();
338            for di in 0..N_DIMS {
339                facebasis.axes[i][di] *= norm;
340            }
341        }
342
343        for ddi in 0..N_DIMS - 1 {
344            facebasis.r[0][ddi] = 0.0;
345        }
346        for vi in 1..nvertices {
347            for ddi in 0..N_DIMS - 1 {
348                let mut dotval = 0.0;
349                for di in 0..N_DIMS {
350                    dotval += vs[vi - 1][di] * facebasis.axes[ddi][di];
351                }
352                facebasis.r[vi][ddi] = dotval;
353            }
354        }
355        facebasis
356    }

358    fn intersect_line_with_face(&self, x: &RVector, dir: &RVector) -> Result<Intersect, RTCError> {
359        let eps_inv = 1.0 / cc::CITRUS_RT_EPS;
360        let mut vs = RMatrix::zeros((N_DIMS - 1, N_DIMS));
361        let mut norm = RVector::zeros(N_DIMS);
362        let mut px_in_face = RVector::zeros(N_DIMS - 1);
363        let mut t_mat = vec![RVector::zeros(N_DIMS - 1); N_DIMS - 1];
364        let mut b_vec = RVector::zeros(N_DIMS - 1);
365        let mut intersect = Intersect::default();
366
367        for vi in 0..(N_DIMS - 1) {
368            for di in 0..N_DIMS {
369                vs[[vi, di]] = self.r[vi + 1][di] - self.r[0][di];
370            }
371        }
372        if N_DIMS == 2 {
373            norm[0] = -vs[[0, 1]];
374            norm[1] = vs[[0, 0]];
375        } else if N_DIMS == 3 {
376            // Calculate norm via cross product.
377            for di in 0..N_DIMS {
378                let j = (di + 1) % N_DIMS;
379                let k = (di + 2) % N_DIMS;
380                norm[di] = vs[[0, j]] * vs[[1, k]] - vs[[0, k]] * vs[[1, j]];
381            }
382        } else {
383            // Calculate norm via SVD
384            let svd_res = vs.svd(false, true).map_err(|_| RTCError::SVDFail)?;
385            let svs = svd_res.1;
386            let svv = svd_res.2.ok_or(RTCError::SVDFail)?;
387
388            let ci = 0;
389            let mut ci_of_max = ci;
390            let mut max_singular_value = svs[ci];
391            for ci in 1..N_DIMS {
392                if svs[ci] > max_singular_value {
393                    ci_of_max = ci;
394                    max_singular_value = svs[ci];
395                }
396            }
397            let mut ci_of_min: isize = -1;
398            for ci in 0..N_DIMS {
399                if ci == ci_of_max {
400                    continue;
401                }
402                let singular_value = svs[ci];
403                if singular_value * eps_inv < max_singular_value {
404                    if ci_of_min >= 0 {
405                        return Err(RTCError::NonSpan);
406                    }
407                    ci_of_min = ci as isize;
408                }
409            }
410            for di in 0..N_DIMS {
411                norm[di] = svv[[di, ci_of_min as usize]];
412            }
413        }
414
415        let mut test_sum_for_clockwise = 0.0;
416        for di in 0..N_DIMS {
417            test_sum_for_clockwise += norm[di] * (self.r[0][di] - self.simplex_centres[di]);
418        }
419        if test_sum_for_clockwise < 0.0 {
420            norm *= -1.0;
421        }
422
423        let norm_dot_dx = (&norm * dir).sum();
424        intersect.orientation = match norm_dot_dx {
425            x if x > 0.0 => Orientation::ExitFace,
426            x if x < 0.0 => Orientation::EntryFace,
427            _ => return Ok(intersect),
428        };
429
430        let mut numerator = 0.0;
431        for di in 0..N_DIMS {
432            numerator += norm[di] * (self.r[0][di] - x[di]);
433        }
434        intersect.dist = numerator / norm_dot_dx;
435        let face_plus_basis = self.basis(N_DIMS);
436        for i in 0..N_DIMS - 1 {
437            px_in_face[i] = 0.0;
438            for di in 0..N_DIMS {
439                px_in_face[i] += (x[di] + intersect.dist * dir[di] - face_plus_basis.origin[di])
440                    * face_plus_basis.axes[i][di];
441            }
442        }
443
444        if N_DIMS == 2 || N_DIMS == 3 {
445            for i in 0..N_DIMS - 1 {
446                for j in 0..N_DIMS - 1 {
447                    t_mat[i][j] = face_plus_basis.r[j + 1][i] - face_plus_basis.r[0][i];
448                    b_vec[i] = px_in_face[i] - face_plus_basis.r[0][i];
449                }
450            }
451            if N_DIMS == 2 {
452                intersect.bary[1] = b_vec[0] / t_mat[0][0];
453            } else {
454                let det = t_mat[0][0] * t_mat[1][1] - t_mat[0][1] * t_mat[1][0];
455                intersect.bary[1] = (t_mat[1][1] * b_vec[0] - t_mat[0][1] * b_vec[1]) / det;
456                intersect.bary[2] = (-t_mat[1][0] * b_vec[0] + t_mat[0][0] * b_vec[1]) / det;
457            }
458        } else {
459            let mut t = RMatrix::zeros((N_DIMS - 1, N_DIMS - 1));
460            let mut b = RVector::zeros(N_DIMS - 1);
461            for i in 0..N_DIMS - 1 {
462                for j in 0..N_DIMS - 1 {
463                    t[[i, j]] = face_plus_basis.r[j + 1][i] - face_plus_basis.r[0][i];
464                }
465                b[i] = px_in_face[i] - face_plus_basis.r[0][i];
466            }
467            let x = t.solve(&b).map_err(|_| RTCError::SolverFail)?;
468
469            for i in 0..N_DIMS - 1 {
470                intersect.bary[i + 1] = x[i];
471            }
472        }
473
474        intersect.bary[0] = 1.0;
475        for i in 1..N_DIMS {
476            intersect.bary[0] -= intersect.bary[i];
477        }
478        let di = 0;
479        if intersect.bary[di] < 0.5 {
480            intersect.coll_par = intersect.bary[di];
481        } else {
482            intersect.coll_par = 1.0 - intersect.bary[di];
483        }
484        for di in 1..N_DIMS {
485            if intersect.bary[di] < 0.5 {
486                if intersect.bary[di] < intersect.coll_par {
487                    intersect.coll_par = intersect.bary[di];
488                }
489            } else if 1.0 - intersect.bary[di] < intersect.coll_par {
490                intersect.coll_par = 1.0 - intersect.bary[di];
491            }
492        }
493
494        Ok(intersect)
495    }
496}

600    let mut num_entry_faces = 0;
601    let mut entry_simplices = Vec::new();
602    let mut entry_fis = Vec::new();
603    let mut entry_intersects = Vec::new();
604
605    for (isimplex, simplex) in rtp.simplices.iter().enumerate() {
606        for fi in 0..num_faces {
607            if simplex.neigh[fi].is_none() {
608                let face = rtp.simplices[isimplex].extract_face(fi, &rtp.vertex_coords);
609                let mut intersect = face.intersect_line_with_face(x, dir)?;
610                intersect.fi = fi;
611                if intersect.orientation == Orientation::EntryFace
612                    && intersect.coll_par + cc::CITRUS_RT_EPS > 0.0
613                {
614                    if num_entry_faces > MAX_NUM_ENTRY_FACES {
615                        return Err(RTCError::TooManyEntries);
616                    }
617                    entry_simplices.push(isimplex);
618                    entry_fis.push(fi);
619                    entry_intersects.push(intersect);
620                    num_entry_faces += 1;
621                }
622            }
623        }
624    }
625
626    if num_entry_faces == 0 {
627        return Ok(RTCResult::NoEntryFaces {
628            entry: Intersect::default(),
629        });
630    }
631
632    let mut i = 0;
633    let mut status = 1;
634    let mut cntxt = ChainContext {
635        cell_visited: vec![false; rtp.simplices.len()],
636        isimplex: entry_simplices[i],
637        entry_face_index: entry_fis[i],
638        ncells_in_chain: 0,
639        len_chain_ptrs: 0,
640        chain_of_cell_ids: Vec::with_capacity(1024),
641        cell_exit_intersects: Vec::with_capacity(1024),
642    };
643    while i < num_entry_faces && status > 0 {
644        status = cntxt.build_ray_cell_chain(x, dir, rtp, rtp.simplices.as_slice())?;
645        i += 1;
646    }
647
648    if status == 0 {
649        Ok(RTCResult::Success {
650            entry_intersect_first_cell: mem::take(&mut entry_intersects[i - 1]),
651            chain_cell_ids: cntxt.chain_of_cell_ids,
652            exit_intersects: cntxt.cell_exit_intersects,
653            len_chain_ptrs: cntxt.len_chain_ptrs,
654        })
655    } else {
656        Ok(RTCResult::FailedToBuildChain {
657            entry: Intersect::default(),
658        })
659    }
660}

701/// thus so can the value of the line-shape function at a given frequency and

705///
706/// # Note
707/// This is called from within the multi-threaded block.
708fn calc_line_amp_sample(
709    vfac: &mut f64,
710    projection_velocities: &RVector,
711    delta_v: f64,
712    binv: f64,
713    nsteps: usize,
714    nsteps_inv: f64,
715) -> f64 {
716    for i in 0..nsteps {
717        let v = delta_v - projection_velocities[i];
718        let val = v.abs() * binv;
719        if val <= 2500.0 {

721        }
722    }

724    *vfac

734/// This is called from within the multi-threaded block.
735fn calc_line_amp_interp(binv: f64, delta_v: f64, projection_vel_ray: f64) -> f64 {
736    const THRESHOLD: f64 = 2500.0;

738    let val = v.abs() * binv;

741    } else {

743    }

747///
748/// The bulk velocity of the model material can vary significantly with position,
749/// which affects the value of the line-shape function at a given frequency and
750/// direction. This function computes `vfac`, an approximate average of the
751/// line-shape function along a path of length `ds` in the direction of the
752/// line of sight.
753///
754/// # Note
755/// This function is invoked within a multi-threaded context.
756fn calc_line_amp_erf(
757    binv: f64,
758    delta_v: f64,
759    projection_vel_old: f64,
760    projection_vel_new: f64,
761) -> f64 {
762    const THRESHOLD: f64 = 1e-4;
763
764    let vb_old = binv * (delta_v - projection_vel_old);
765    let vb_new = binv * (delta_v - projection_vel_new);
766    if vb_new - vb_old > (2.0 * THRESHOLD) {
767        get_erf(vb_old, vb_new)
768    } else {
769        gauss_line((0.5) * (vb_new + vb_old), 1.0)
770    }

773/// Computes the distance to the next Voronoi face in a given direction.
774///
775/// Returns `ds`, the (always positive) distance from the current position `x`
776/// to the next Voronoi face in the direction of vector `dx`, and `nposn`,
777/// the ID of the neighboring grid cell adjacent to that face.
778///
779/// # Note
780/// This function is invoked within a multi-threaded context.
781fn line_plane_intersect(
782    grid: &Grid,
783    x: &RVector,
784    dx: &RVector,
785    ds: f64,
786    cutoff: f64,
787) -> (f64, usize) {
788    for i in 0..grid.num_neigh {
789        let numerator = (grid.x[0] + grid.dir[i].x[0] / 2.0 - x[0]) * grid.dir[i].x[0]
790            + (grid.x[1] + grid.dir[i].x[1] / 2.0 - x[1]) * grid.dir[i].x[1]
791            + (grid.x[2] + grid.dir[i].x[2] / 2.0 - x[2]) * grid.dir[i].x[2];
792        let denominator =
793            dx[0] * grid.dir[i].x[0] + dx[1] * grid.dir[i].x[1] + dx[2] * grid.dir[i].x[2];
794        if denominator.abs() > 0.0 {
795            let newdist = numerator / denominator;
796            if newdist < ds && newdist > cutoff {
797                if let Some(neigh) = &grid.neigh[i] {
798                    return (newdist, neigh.id as usize);
799                }
800            }
801        }
802    }
803    (ds, 0)
804}

806/// Evaluates the light intensity along a line of sight for a given image pixel.
807///
808/// For a specified pixel position, this function computes the total light
809/// emitted or absorbed along the corresponding line of sight through the
810/// (potentially rotated) model. The computation is performed across multiple
811/// frequenciesone per channel in the output image.
812///
813/// The algorithm follows a similar approach to `calculateJBar()`, which determines
814/// the average radiant flux impinging on a grid cell. A notional photon is
815/// initiated at the model side nearest to the observer and propagated in the
816/// receding direction until it reaches the far side.
817///
818/// While this approach is conceptually non-physical, it simplifies the computation.
819///
820/// # Note
821/// This function is invoked within a multi-threaded context.
822fn trace_ray(
823    ray: &mut RayData,
824    gp: &[Grid],
825    img: &Image,
826    par: &Parameters,
827    mol_data: &[MolData],
828) -> Result<(), RTCError> {
829    const N_STEPS_THROUGH_CELL: usize = 10;
830    const N_STEPS_INV: f64 = 1.0 / (N_STEPS_THROUGH_CELL as f64);
831    let mut projection_velocities = RVector::zeros(N_STEPS_THROUGH_CELL);
832
833    let cutoff = par.min_scale * 1.0e-7;
834    let xp = ray.x;
835    let yp = ray.y;
836    if xp * xp + yp * yp > par.radius_squ {
837        return Ok(());
838    }
839
840    let mut x = RVector::zeros(N_DIMS);
841    let mut dx = RVector::zeros(N_DIMS);
842
843    let zp = -(par.radius_squ - (xp * xp + yp * yp)).sqrt();
844    for di in 0..N_DIMS {
845        x[di] = xp * img.rotation_matrix[[di, 0]]
846            + yp * img.rotation_matrix[[di, 1]]
847            + zp * img.rotation_matrix[[di, 2]];
848        dx[di] = img.rotation_matrix[[di, 2]];
849    }
850
851    let mut dist2 = (x[0] - gp[0].x[0]) * (x[0] - gp[0].x[0])
852        + (x[1] - gp[0].x[1]) * (x[1] - gp[0].x[1])
853        + (x[2] - gp[0].x[2]) * (x[2] - gp[0].x[2]);
854    let mut posn = 0;
855    for (i, gpi) in gp.iter().enumerate().take(par.ncell).skip(1) {
856        let ndist2 = (x[0] - gpi.x[0]) * (x[0] - gpi.x[0])
857            + (x[1] - gpi.x[1]) * (x[1] - gpi.x[1])
858            + (x[2] - gpi.x[2]) * (x[2] - gpi.x[2]);
859        if ndist2 < dist2 {
860            posn = i;
861            dist2 = ndist2;
862        }
863    }
864    let mut col = 0.0;
865    while col < 2.0 * zp.abs() {
866        let (ds, nposn) = line_plane_intersect(&gp[posn], &x, &dx, -2.0 * zp - col, cutoff);
867        if par.polarization {
868            let (snu_pol, alpha) =
869                source_fn_polarized(&gp[posn].mag_field, &gp[posn].cont, &img.rotation_matrix)?;
870            let dtau = alpha * ds;
871            let (mut remnant_snu, _) = calc_source_fn(dtau, par.taylor_cutoff);
872            remnant_snu *= ds;
873            for (stokesi, snu_poli) in snu_pol.iter().enumerate().take(img.nchan) {
874                let brightness_increment = (-ray.tau[stokesi]).exp() * remnant_snu * snu_poli;
875                ray.intensity[stokesi] += brightness_increment;
876                ray.tau[stokesi] += dtau;
877            }
878        } else if img.do_line && par.use_vel_func_in_raytrace {
879            for i in 0..N_STEPS_THROUGH_CELL {
880                let d = i as f64 * ds * N_STEPS_INV;
881                // let vel = velocity(x[0] + (dx[0] * d), x[1] + (dx[1] * d), x[2] + (dx[2] * d));
882                // projection_velocities[i] = dx.dot(&vel);
883            }
884            let mut cont_jnu = 0.0;
885            let mut cont_alpha = 0.0;
886            (cont_jnu, cont_alpha) = source_fn_cont(cont_jnu, cont_alpha, &gp[posn].cont);
887            for ichan in 0..img.nchan {
888                let mut jnu = cont_jnu;
889                let mut alpha = cont_alpha;
890                let v_this_chan = (ichan as f64 - (img.nchan - 1) as f64 * 0.5) * img.vel_res;
891                if img.do_line {
892                    for moli in 0..par.n_species {
893                        for linei in 0..mol_data[moli].nline {
894                            if mol_data[moli].freq[linei] > img.freq - img.bandwidth * 0.5
895                                && mol_data[moli].freq[linei] < img.freq + img.bandwidth * 0.5
896                            {
897                                let line_red_shift = if img.trans > -1 {
898                                    (mol_data[moli].freq[img.trans as usize]
899                                        - mol_data[moli].freq[linei])
900                                        / mol_data[moli].freq[img.trans as usize]
901                                        * cc::SPEED_OF_LIGHT_SI
902                                } else {
903                                    (img.freq - mol_data[moli].freq[linei])
904                                        / mol_data[moli].freq[linei]
905                                        * cc::SPEED_OF_LIGHT_SI
906                                };
907                                let delta_v = v_this_chan - img.source_velocity - line_red_shift;
908                                let binv = if let Some(mols) = &gp[posn].mol {
909                                    mols[moli].binv
910                                } else {
911                                    // This should never happen
912                                    return Err(RTCError::EmptyGrid);
913                                };
914                                let vfac = if par.use_vel_func_in_raytrace {
915                                    calc_line_amp_sample(
916                                        &mut 0f64,
917                                        &projection_velocities,
918                                        delta_v,
919                                        binv,
920                                        N_STEPS_THROUGH_CELL,
921                                        N_STEPS_INV,
922                                    )
923                                } else {
924                                    gauss_line(delta_v - dx.dot(&gp[posn].vel), binv)
925                                };
926                                if let Some(mols) = &gp[posn].mol {
927                                    (jnu, alpha) = source_fn_line(
928                                        &mols[moli],
929                                        &mol_data[moli],
930                                        vfac,
931                                        linei,
932                                        jnu,
933                                        alpha,
934                                    );
935                                } else {
936                                    return Err(RTCError::EmptyGrid);
937                                };
938                            }
939                        }
940                    }
941                }
942                let dtau = alpha * ds;
943                let (mut remnant_snu, _) = calc_source_fn(dtau, par.taylor_cutoff);
944                remnant_snu *= jnu * ds;
945                let brightness_increment = (-ray.tau[ichan]).exp() * remnant_snu;
946                ray.intensity[ichan] += brightness_increment;
947                ray.tau[ichan] += dtau;
948            }
949        }
950
951        for di in 0..N_DIMS {
952            x[di] += ds * dx[di];
953        }
954        col += ds;
955        posn = nposn;
956    }
957    Ok(())
958}

960/// Performs linear interpolation over a simplex (triangle) using barycentric coordinates.
961///
962/// This function takes:
963/// 1. `N` values `V_i` at the vertices of a simplex (triangle),
964/// 2. The barycentric coordinates of a point within the simplex,
965///    and returns the interpolated value at that point.
966///
967/// This approach is equivalent to using linear shape functions in Finite Element Analysis.
968/// The interpolation uses shape functions `Q_i`, where each shape function:
969/// - Is zero at all vertices except the `i`-th,
970/// - Has value one at the `i`-th vertex.
971///
972/// The interpolated value at point `r_` is computed as:
973///
974/// ```
975///           N
976///           
977/// f(r_) =      V_i * Q_i(r_)
978///           
979///          i=1
980/// ```
981///
982/// For linear interpolation, each shape function `Q_i(r_)` is equal to the `i`-th barycentric coordinate `B_i` of `r_`.
983///
984/// In this case, `N == 3`; the simplex is a triangular face of a Delaunay cell,
985/// and the interpolation point is the intersection of a ray with that face.
986/// This technique is used to interpolate several grid-based quantities.
987///
988/// For further reference, see the Wikipedia article on [Barycentric coordinates](https://en.wikipedia.org/wiki/Barycentric_coordinate_system).
989///
990/// # Note
991/// This is called from within the multi-threaded block.
992fn do_barycentric_interpolation(
993    gip: &mut GridInterp,
994    gp: &[Grid],
995    mol_data: &[MolData],
996    par: &Parameters,
997    gis: &[usize; 3],
998    intersect: &Intersect,
999    x_cmpts_ray: RVector,
1000) {
1001    gip.x_component_ray = intersect.bary.dot(&x_cmpts_ray);
1002    for di in 0..N_DIMS {
1003        let vals: RVector = gis.iter().map(|&i| gp[i].x[di]).collect();
1004        gip.x[di] = vals.dot(&intersect.bary);
1005    }
1006    for di in 0..3 {
1007        let vals: RVector = gis.iter().map(|&i| gp[i].mag_field[di]).collect();
1008        gip.magnetic_field[di] = vals.dot(&intersect.bary);
1009    }
1010    for (moli, mdi) in mol_data.iter().enumerate().take(par.n_species) {
1011        let binvals: RVector = gis
1012            .iter()
1013            .filter_map(|&i| {
1014                gp[i]
1015                    .mol
1016                    .as_ref()
1017                    .and_then(|mol_vec| mol_vec.get(moli).map(|m| m.binv))
1018            })
1019            .collect();
1020        gip.mol[moli].binv = binvals.dot(&intersect.bary);
1021        for leveli in 0..mdi.nlev {
1022            let vals: RVector = gis
1023                .iter()
1024                .filter_map(|&i| {
1025                    gp[i]
1026                        .mol
1027                        .as_ref()
1028                        .and_then(|mol_vec| mol_vec.get(moli).map(|m| m.spec_num_dens[leveli]))
1029                })
1030                .collect();
1031            gip.mol[moli].spec_num_dens[leveli] = vals.dot(&intersect.bary);
1032        }
1033    }
1034    let dustvals: RVector = gis.iter().map(|&i| gp[i].cont.dust).collect();
1035    gip.cont.dust = dustvals.dot(&intersect.bary);
1036    let knuvals: RVector = gis.iter().map(|&i| gp[i].cont.knu).collect();
1037    gip.cont.knu = knuvals.dot(&intersect.bary);
1038}

1040fn do_segment_interpolation(
1041    gips: &mut [GridInterp],
1042    par: &Parameters,
1043    mol_data: &[MolData],
1044    ia: usize,
1045    si: usize,
1046    n_seg_inv: f64,
1047) {
1048    let frac_a = (si as f64 + 0.5) * n_seg_inv;
1049    let frac_b = 1.0 - frac_a;
1050    let ib = 1 - ia;
1051
1052    gips[2].x_component_ray = frac_a * gips[ib].x_component_ray + frac_b * gips[ia].x_component_ray;
1053
1054    for di in 0..N_DIMS {
1055        gips[2].x[di] = frac_a * gips[ib].x[di] + frac_b * gips[ia].x[di];
1056    }
1057    for di in 0..3 {
1058        gips[2].magnetic_field[di] =
1059            frac_a * gips[ib].magnetic_field[di] + frac_b * gips[ia].magnetic_field[di];
1060    }
1061
1062    for (moli, mdi) in mol_data.iter().enumerate().take(par.n_species) {
1063        gips[2].mol[moli].binv =

1070    }

1076fn calc_second_order_shape_functions(
1077    buffer: &mut BaryVelocityBuffer,
1078    rayi: usize,
1079) -> Result<(), RTCError> {
1080    let barys = match rayi {
1081        0 => &buffer.entry_cell_bary,
1082        1 => &buffer.exit_cell_bary,
1083        2 => &buffer.mid_cell_bary,
1084        _ => return Err(RTCError::Other("Bad ray index".to_string())),
1085    };
1086
1087    let mut counter = 0;
1088    for vi in 0..buffer.num_vertices {
1089        buffer.shape_fns[counter] = barys[vi] * (2.0 * barys[vi] - 1.0);
1090        counter += 1;
1091    }
1092
1093    for ei in 0..buffer.num_edges {
1094        buffer.shape_fns[counter] = 4.0
1095            * barys[buffer.edge_vertex_indices[ei][0]]
1096            * (barys[buffer.edge_vertex_indices[ei][1]]);
1097        counter += 1;
1098    }
1099
1100    Ok(())
1101}

1104    vels: &mut RVector,
1105    buffer: &BaryVelocityBuffer,
1106) {

1108    let mut counter = 0;
1109    for vi in 0..buffer.num_vertices {
1110        for (di, vel) in vels.iter_mut().enumerate().take(N_DIMS) {
1111            *vel += buffer.vertex_velocities[vi][di] * buffer.shape_fns[counter];
1112        }
1113        counter += 1;
1114    }
1115    for ei in 0..buffer.num_edges {
1116        for (di, vel) in vels.iter_mut().enumerate().take(N_DIMS) {
1117            *vel += buffer.edge_velocities[ei][di] * buffer.shape_fns[counter];
1118        }
1119        counter += 1;
1120    }
1121}

1145/// For a given image pixel position, this function evaluates the intensity of the
1146/// total light emitted/absorbed along that line of sight through the (possibly
1147/// rotated) model. The calculation is performed for several frequencies, one per
1148/// channel of the output image.
1149///
1150/// Note that the algorithm employed here to solve the RTE is similar to that
1151/// employed in the function calculateJBar() which calculates the average radiant
1152/// flux impinging on a grid cell: namely the notional photon is started at the side
1153/// of the model near the observer and 'propagated' in the receding direction until
1154/// it 'reaches' the far side. This is rather non-physical in conception but it
1155/// makes the calculation easier.
1156///
1157/// This version of traceray implements a new algorithm in which the population
1158/// values are interpolated linearly from those at the vertices of the Delaunay cell
1159/// which the working point falls within.
1160///
1161/// A note about the object 'gips': this is an array with 3 elements, each one a
1162/// struct of type 'gridInterp'. This struct is meant to store as many of the
1163/// grid-point quantities (interpolated from the appropriate values at actual grid
1164/// locations) as are necessary for solving the radiative transfer equations along
1165/// the ray path. The first 2 entries give the values for the entry and exit points
1166/// to a Delaunay cell, but which is which can change, and is indicated via the
1167/// variables entryI and exitI (this is a convenience to avoid copying the values,
1168/// since the values for the exit point of one cell are obviously just those for
1169/// entry point of the next). The third entry stores values interpolated along the
1170/// ray path within a cell.
1171///
1172/// # Note
1173/// This is called from within the multi-threaded block.
1174fn trace_ray_smooth(
1175    ray: &mut RayData,
1176    gips: &mut [GridInterp],
1177    gp: &[Grid],
1178    img: &Image,
1179    par: &Parameters,
1180    mol_data: &[MolData],
1181    rtp: &RTPreparation,
1182) -> Result<(), RTCError> {
1183    const NUM_SEGMENTS: usize = 5;
1184    const N_SEGMENTS_INV: f64 = 1.0 / (NUM_SEGMENTS as f64);
1185
1186    const NUM_FACES: usize = N_DIMS + 1;
1187    const N_VERT_PER_FACE: usize = 3;
1188    const NUM_RAY_INTERP_SAMPLE: usize = 3;
1189
1190    for ichan in 0..img.nchan {
1191        ray.tau[ichan] = 0.0;
1192        ray.intensity[ichan] = 0.0;
1193    }
1194
1195    let xp = ray.x;
1196    let yp = ray.y;
1197
1198    if (xp * xp + yp * yp) > par.radius_squ {
1199        return Ok(());
1200    }
1201
1202    let mut x = RVector::zeros(N_DIMS);
1203    let mut dir = RVector::zeros(N_DIMS);
1204
1205    let zp = -(par.radius_squ - (xp * xp + yp * yp)).sqrt();
1206    for di in 0..N_DIMS {
1207        x[di] = xp * img.rotation_matrix[[di, 0]]
1208            + yp * img.rotation_matrix[[di, 1]]
1209            + zp * img.rotation_matrix[[di, 2]];
1210        dir[di] = img.rotation_matrix[[di, 2]];
1211    }
1212
1213    let status = follow_ray_through_cells(N_DIMS, &x, &dir, rtp)?;
1214    let (entry_intersect_first_cell, chain_cell_ids, exit_intersects, len_chain_ptrs) = match status
1215    {
1216        RTCResult::Success {
1217            entry_intersect_first_cell,
1218            chain_cell_ids,
1219            exit_intersects,
1220            len_chain_ptrs,
1221        } => (
1222            entry_intersect_first_cell,
1223            chain_cell_ids,
1224            exit_intersects,
1225            len_chain_ptrs,
1226        ),
1227        _ => return Ok(()),
1228    };
1229
1230    let mut gis = [[0; N_VERT_PER_FACE]; 2];
1231
1232    let (inter_cell_keys, mut vel_buffer): (Vec<InterCellKey>, Option<BaryVelocityBuffer>) =
1233        if img.do_line && img.do_interpolate_vels {
1234            let vel_buffer = BaryVelocityBuffer::new();
1235            let mut inter_cell_keys = Vec::with_capacity(len_chain_ptrs - 1);
1236            let mut dci1 = chain_cell_ids[0];
1237            for ci in 0..len_chain_ptrs - 1 {
1238                let dci0 = dci1;
1239                dci1 = chain_cell_ids[ci + 1];
1240                let mut vvi = 0;
1241                for fi in 0..NUM_FACES {
1242                    if fi != exit_intersects[ci].fi {
1243                        gis[0][vvi] = rtp.simplices[dci0].vertex[fi];
1244                        vvi += 1;
1245                    }
1246                }
1247                inter_cell_keys.push(InterCellKey {
1248                    fi_entered_cell: -1,
1249                    ..Default::default()
1250                });
1251
1252                let mut vvi = 0;
1253                for vi in 0..vel_buffer.num_vertices {
1254                    let trial_gi = rtp.simplices[dci1].vertex[vi];
1255                    let mut match_found = false;
1256                    let mut fi = 0;
1257                    for i in 0..N_VERT_PER_FACE {
1258                        if trial_gi == gis[0][i] {
1259                            match_found = true;
1260                            fi = i;
1261                            break;
1262                        }
1263                    }
1264                    if match_found {
1265                        inter_cell_keys[ci].exited_face_ids[vvi] = fi;
1266                        vvi += 1;
1267                    } else {
1268                        if inter_cell_keys[ci].fi_entered_cell != -1 {
1269                            return Err(RTCError::OppositeVertexAmbiguity);
1270                        }
1271                        inter_cell_keys[ci].fi_entered_cell = vi as i32;
1272                    }
1273                }
1274            }
1275            (inter_cell_keys, Some(vel_buffer))
1276        } else {
1277            (Vec::new(), None)
1278        };
1279
1280    let mut entry_index = 0;
1281    let mut exit_index = 1;
1282    let dci = chain_cell_ids[0];
1283
1284    let mut vvi = 0;
1285    for fi in 0..NUM_FACES {
1286        if fi != entry_intersect_first_cell.fi {
1287            gis[entry_index][vvi] = rtp.simplices[dci].vertex[fi];
1288            vvi += 1;
1289        }
1290    }
1291
1292    let mut x_components_ray = Vec::with_capacity(N_VERT_PER_FACE);
1293
1294    for vi in 0..N_VERT_PER_FACE {
1295        let mut gpis = Vec::with_capacity(N_DIMS);
1296        for val in gp[gis[entry_index][vi]].x.iter() {
1297            gpis.push(*val);
1298        }
1299        let gpis = RVector::from_vec(gpis);
1300        x_components_ray.push(dir.dot(&gpis));
1301    }
1302
1303    do_barycentric_interpolation(
1304        &mut gips[entry_index],
1305        gp,
1306        mol_data,
1307        par,
1308        &gis[entry_index],
1309        &entry_intersect_first_cell,
1310        RVector::from_vec(x_components_ray),
1311    );
1312
1313    let mut do_ray = [true; NUM_RAY_INTERP_SAMPLE];
1314    let mut ray_velocities: [RVector; NUM_RAY_INTERP_SAMPLE] =
1315        array::from_fn(|_| RVector::zeros(N_DIMS));
1316
1317    let mut projection_ray_velocities = RVector::zeros(NUM_RAY_INTERP_SAMPLE);
1318
1319    for ci in 0..len_chain_ptrs {
1320        let dci = chain_cell_ids[ci];
1321        let mut vvi = 0;
1322        for fi in 0..NUM_FACES {
1323            if fi != exit_intersects[ci].fi {
1324                gis[exit_index][vvi] = rtp.simplices[dci].vertex[fi];
1325                vvi += 1;
1326            }
1327        }
1328        let mut x_components_ray = Vec::with_capacity(N_VERT_PER_FACE);
1329
1330        for vi in 0..N_VERT_PER_FACE {
1331            let mut gpis = Vec::with_capacity(N_DIMS);
1332            for val in gp[gis[exit_index][vi]].x.iter() {
1333                gpis.push(*val);
1334            }
1335            let gpis = RVector::from_vec(gpis);
1336            x_components_ray.push(dir.dot(&gpis));
1337        }
1338        do_barycentric_interpolation(
1339            &mut gips[exit_index],
1340            gp,
1341            mol_data,
1342            par,
1343            &gis[exit_index],
1344            &exit_intersects[ci],
1345            RVector::from_vec(x_components_ray),
1346        );
1347
1348        let mut projection_vel_old = 0.0;
1349        let mut projection_vel_offset = 0.0;
1350        let mut projection_vel_new = 0.0;
1351        let mut projection_vel_ray = 0.0;
1352
1353        if img.do_line && img.do_interpolate_vels {
1354            if let Some(ref mut buffer) = vel_buffer {
1355                for vi in 0..buffer.num_vertices {
1356                    let gi = rtp.simplices[dci].vertex[vi];
1357                    for di in 0..N_DIMS {
1358                        buffer.vertex_velocities[vi][di] = gp[gi].vel[di];
1359                    }
1360                }
1361                for ei in 0..buffer.num_edges {
1362                    let gi0 = rtp.simplices[dci].vertex[buffer.edge_vertex_indices[ei][0]];
1363                    let gi1 = rtp.simplices[dci].vertex[buffer.edge_vertex_indices[ei][1]];
1364                    let mut neighbor_not_found = true;
1365                    let mut edge_index = 0;
1366                    for k in 0..gp[gi0].num_neigh {
1367                        if let Some(neigh) = &gp[gi0].neigh[k] {
1368                            if neigh.id == gi1 as i32 {
1369                                neighbor_not_found = false;
1370                                edge_index = k;
1371                                break;
1372                            }
1373                        }
1374                    }
1375                    if neighbor_not_found {
1376                        return Err(RTCError::NeighborNotFound);
1377                    }
1378                    for di in 0..N_DIMS {
1379                        buffer.edge_velocities[ei][di] = gp[gi0].v2[3 * edge_index + di];
1380                    }
1381                }
1382                if ci == 0 {
1383                    let mut vvi = 0;
1384                    for vi in 0..buffer.num_vertices {
1385                        if vi == entry_intersect_first_cell.fi {
1386                            buffer.entry_cell_bary[vi] = 0.0;
1387                        } else {
1388                            buffer.entry_cell_bary[vi] = entry_intersect_first_cell.bary[vvi];
1389                            vvi += 1;
1390                        }
1391                    }
1392                    do_ray[0] = true;
1393                    do_ray[1] = true;
1394                    do_ray[2] = true;
1395                } else {
1396                    let mut vvi = 0;
1397                    for vi in 0..buffer.num_vertices {
1398                        if vi == inter_cell_keys[ci - 1].fi_entered_cell as usize {
1399                            buffer.entry_cell_bary[vi] = 0.0;
1400                        } else {
1401                            buffer.entry_cell_bary[vi] = exit_intersects[ci - 1].bary
1402                                [inter_cell_keys[ci - 1].exited_face_ids[vvi]];
1403                            vvi += 1;
1404                        }
1405                    }
1406                    do_ray[0] = false;
1407                    do_ray[1] = true;
1408                    do_ray[2] = true;
1409                    for di in 0..N_DIMS {
1410                        ray_velocities[0][di] = ray_velocities[1][di];
1411                    }
1412                }
1413                let mut vvi = 0;
1414                for vi in 0..buffer.num_vertices {
1415                    if vi == exit_intersects[ci].fi {
1416                        buffer.exit_cell_bary[vi] = 0.0;
1417                    } else {
1418                        buffer.exit_cell_bary[vi] = exit_intersects[ci].bary[vvi];
1419                        vvi += 1;
1420                    }
1421                    buffer.mid_cell_bary[vi] =
1422                        0.5 * (buffer.entry_cell_bary[vi] + buffer.exit_cell_bary[vi]);
1423                }
1424                do_barycentric_interpolations_vel::<N_DIMS, N_VERT_PER_FACE>(
1425                    buffer,
1426                    &mut ray_velocities,
1427                    &do_ray,
1428                )?;
1429                for i in 0..NUM_RAY_INTERP_SAMPLE {
1430                    projection_ray_velocities[i] = dir.dot(&ray_velocities[i]);
1431                }
1432                projection_vel_old =
1433                    do_segment_interpolation_scalar(&projection_ray_velocities, 0.0);
1434                let projection_vel_second_deriv = (projection_ray_velocities[0]
1435                    + projection_ray_velocities[1]
1436                    - 2.0 * projection_ray_velocities[2])
1437                    * 4.0;
1438                projection_vel_offset =
1439                    -projection_vel_second_deriv * N_SEGMENTS_INV * N_SEGMENTS_INV / 6.0;
1440            }
1441        }
1442        let ds =
1443            (gips[exit_index].x_component_ray - gips[entry_index].x_component_ray) * N_SEGMENTS_INV;
1444        for si in 0..NUM_SEGMENTS {
1445            do_segment_interpolation(gips, par, mol_data, entry_index, si, N_SEGMENTS_INV);
1446            if par.polarization {
1447                let (snu_pol, alpha) = source_fn_polarized(
1448                    &gips[2].magnetic_field,
1449                    &gips[2].cont,
1450                    &img.rotation_matrix,
1451                )?;
1452                let dtau = alpha * ds;
1453                let (mut remnant_snu, _) = calc_source_fn(dtau, par.taylor_cutoff);
1454                remnant_snu *= ds;
1455                for (stokesi, snu_poli) in snu_pol.iter().enumerate().take(img.nchan) {
1456                    let brightness_increment = (-ray.tau[stokesi]).exp() * remnant_snu * snu_poli;
1457                    ray.intensity[stokesi] += brightness_increment;
1458                    ray.tau[stokesi] += dtau;
1459                }
1460            } else if img.do_line {
1461                if img.do_interpolate_vels {
1462                    projection_vel_new = do_segment_interpolation_scalar(
1463                        &projection_ray_velocities,
1464                        (si as f64 + 1.0) * (N_SEGMENTS_INV),
1465                    );
1466                } else {
1467                    // let vel = velocity(gips[2].x[0], gips[2].x[1], gips[2].x[2]);
1468                    // projection_vel_ray = dir.dot(&vel);
1469                }
1470            }
1471            let cont_jnu = 0.0;
1472            let cont_alpha = 0.0;
1473            let (cont_jnu, cont_alpha) = source_fn_cont(cont_jnu, cont_alpha, &gips[2].cont);
1474            for ichan in 0..img.nchan {
1475                let ichanf = ichan as f64;
1476                let nchanf = img.nchan as f64;
1477                let mut jnu = cont_jnu;
1478                let mut alpha = cont_alpha;
1479                let v_this_channel = (ichanf - (nchanf - 1.0) * 0.5) * img.vel_res;
1480                if img.do_line {
1481                    for (moli, mdi) in mol_data.iter().enumerate().take(par.n_species) {
1482                        for linei in 0..mdi.nline {
1483                            if mdi.freq[linei] > img.freq - img.bandwidth * 0.5
1484                                && mdi.freq[linei] < img.freq + img.bandwidth * 0.5
1485                            {
1486                                let line_red_shift = if img.trans > -1 {
1487                                    (mdi.freq[img.trans as usize] - mdi.freq[linei])
1488                                        / mdi.freq[img.trans as usize]
1489                                        * cc::SPEED_OF_LIGHT_SI
1490                                } else {
1491                                    (img.freq - mdi.freq[linei]) / mdi.freq[linei]
1492                                        * cc::SPEED_OF_LIGHT_SI
1493                                };
1494                                let delta_v = v_this_channel - img.source_velocity - line_red_shift;
1495                                let vfac = if img.do_interpolate_vels {
1496                                    calc_line_amp_erf(
1497                                        projection_vel_old,
1498                                        projection_vel_new,
1499                                        gips[2].mol[moli].binv,
1500                                        delta_v - projection_vel_offset,
1501                                    )
1502                                } else {
1503                                    calc_line_amp_interp(
1504                                        gips[2].mol[moli].binv,
1505                                        delta_v,
1506                                        projection_vel_ray,
1507                                    )
1508                                };
1509                                (jnu, alpha) = source_fn_line(
1510                                    &gips[2].mol[moli],
1511                                    mdi,
1512                                    vfac,
1513                                    linei,
1514                                    jnu,
1515                                    alpha,
1516                                );
1517                            }
1518                        }
1519                    }
1520                }
1521                let dtau = alpha * ds;
1522                let (mut remnant_snu, _) = calc_source_fn(dtau, par.taylor_cutoff);
1523                remnant_snu *= jnu * ds;
1524                let brightness_increment = (-ray.tau[ichan]).exp() * remnant_snu;
1525                ray.intensity[ichan] += brightness_increment;
1526                ray.tau[ichan] += dtau;
1527            }
1528        }
1529        entry_index = exit_index;
1530        exit_index = 1 - exit_index;
1531    }
1532
1533    Ok(())
1534}

1536fn locate_ray_on_image(
1537    x: &[f64; 2],
1538    size: f64,
1539    img_centre_x_pxls: f64,
1540    img_centre_y_pxls: f64,
1541    img: &Image,
1542) -> (bool, usize) {
1543    let xi = (x[0] / size + img_centre_x_pxls).floor() as i64;
1544    let yi = (x[1] / size + img_centre_y_pxls).floor() as i64;
1545
1546    if xi < 0 || xi >= img.pxls || yi < 0 || yi >= img.pxls {
1547        (false, 0)
1548    } else {
1549        (true, (yi * img.pxls + xi) as usize)
1550    }
1551}

1589}

1591fn calc_triangular_barycentric_coords(vertices: &[[f64; 2]; 3], x: f64, y: f64) -> RVector {
1592    let mat = [
1593        [
1594            vertices[0][0] - vertices[2][0],
1595            vertices[1][0] - vertices[2][0],
1596        ],
1597        [
1598            vertices[0][1] - vertices[2][1],
1599            vertices[1][1] - vertices[2][1],
1600        ],
1601    ];
1602    let vec = [x - vertices[2][0], y - vertices[2][1]];
1603    let det = mat[0][0] * mat[1][1] - mat[0][1] * mat[1][0];
1604    let b1 = (mat[1][1] * vec[0] - mat[0][1] * vec[1]) / det;
1605    let b2 = (-mat[1][0] * vec[0] + mat[0][0] * vec[1]) / det;
1606    let b3 = 1.0 - b1 - b2;
1607    RVector::from_vec(vec![b1, b2, b3])
1608}

1620fn convert_cell_to_simplex(cells: &[Cell], par: &Parameters, gp: &[Grid]) -> Vec<Simplex> {
1621    let mut simplices = Vec::with_capacity(par.ncell);
1622
1623    // Step 1: Initialize all simplices with geometry and vertex data
1624    for (icell, cell) in cells.iter().enumerate() {
1625        let mut simplex = Simplex {
1626            id: icell,
1627            neigh: vec![None; N_DIMS + 1],
1628            ..Default::default()
1629        };
1630
1631        for vi in 0..N_DIMS + 1 {
1632            simplex.vertex[vi] = cell.vertex[vi].as_ref().map(|grid| grid.id).unwrap_or(0) as usize;
1633
1634            let gi = simplex.vertex[vi];
1635            for di in 0..N_DIMS {
1636                simplex.centres[di] += gp[gi].x[di];
1637            }
1638        }
1639
1640        for di in 0..N_DIMS {
1641            simplex.centres[di] *= 1.0 / (N_DIMS + 1) as f64;
1642        }
1643
1644        simplices.push(simplex);
1645    }
1646
1647    // Step 2: Assign neighbor indices
1648    for (icell, cell) in cells.iter().enumerate() {
1649        for vi in 0..N_DIMS + 1 {
1650            match cell.neigh[vi] {
1651                Some(neighbor_cell_index) => {
1652                    simplices[icell].neigh[vi] = Some(neighbor_cell_index);
1653                }
1654                None => {
1655                    simplices[icell].neigh[vi] = None;
1656                }
1657            }
1658        }
1659    }
1660
1661    simplices
1662}

1664fn get_2d_cells(rays: &[RayData], num_active_rays: usize) -> Result<Vec<Simplex>> {
1665    const N_DIMS_2D: usize = 2;
1666    const N_FACES_2D: usize = N_DIMS_2D + 1;
1667    const N_FACES_2D_INV: f64 = 1.0 / (N_FACES_2D as f64);
1668
1669    let mut pt_array = Vec::with_capacity(N_DIMS_2D * num_active_rays);
1670    for ray in rays.iter().take(num_active_rays) {
1671        pt_array.push(ray.x);
1672        pt_array.push(ray.y);
1673    }
1674
1675    let pt_array_clone = pt_array.clone();
1676
1677    let qh = QhBuilder::default()
1678        .delaunay(true)
1679        .scale_last(true)
1680        .triangulate(true)
1681        .build_managed(N_DIMS, pt_array)
1682        .map_err(|e| anyhow!("Failed to build Qhull: {:?}", e))?;
1683
1684    let mut num_cells: usize = 0;
1685    for facet in qh.all_facets() {
1686        if !facet.upper_delaunay() {
1687            num_cells += 1;
1688        }
1689    }
1690
1691    let mut simplices_2d: Vec<Simplex> = Vec::with_capacity(num_cells);
1692
1693    let mut fi = 0;
1694    for facet in qh.all_facets() {
1695        if !facet.upper_delaunay() {
1696            simplices_2d[fi].id = facet.id() as usize;
1697            fi += 1;
1698        }
1699    }
1700
1701    let mut fi = 0;
1702    for facet in qh.all_facets() {
1703        if !facet.upper_delaunay() {
1704            let neighbors = facet
1705                .neighbors()
1706                .ok_or_else(|| anyhow!("Failed to get neighbors"))?;
1707            for (i, neighbor) in neighbors.iter().enumerate() {
1708                if neighbor.upper_delaunay() {
1709                    simplices_2d[fi].neigh[i] = None;
1710                } else {
1711                    let mut ffi = 0;
1712                    let mut neighbor_not_found = true;
1713                    while ffi < num_cells && neighbor_not_found {
1714                        if simplices_2d[ffi].id == neighbor.id() as usize {
1715                            simplices_2d[fi].neigh[i] = Some(ffi);
1716                            neighbor_not_found = false;
1717                        }
1718                        ffi += 1;
1719                    }
1720                    if ffi >= num_cells && neighbor_not_found {
1721                        bail!("Something went wrong with the Delaunay triangulation");
1722                    }
1723                }
1724            }
1725            let mut i = 0;
1726            let vertices = facet
1727                .vertices()
1728                .ok_or_else(|| anyhow!("Failed to get vertices"))?;
1729            for vertex in vertices.iter() {
1730                let point_id_this = vertex.point_id(&qh);
1731                match point_id_this {
1732                    Ok(point_id_this) => {
1733                        simplices_2d[fi].vertex[i] = point_id_this as usize;
1734                        i += 1;
1735                    }
1736                    Err(_) => {
1737                        bail!("Failed to get point ID");
1738                    }
1739                }
1740            }
1741            fi += 1;
1742        }
1743    }
1744
1745    for (isimplex_2d, simplex_2d) in simplices_2d.iter_mut().enumerate().take(num_cells) {
1746        for di in 0..N_DIMS_2D {
1747            let mut sum = 0.0;
1748            for vi in 0..N_FACES_2D {
1749                let id = simplex_2d.vertex[vi];
1750                sum += pt_array_clone[id * N_DIMS + di];
1751            }
1752            simplex_2d.centres[di] = sum / N_FACES_2D_INV;
1753        }
1754        simplex_2d.id = isimplex_2d;
1755    }
1756    Ok(simplices_2d)
1757}

1759fn prepare_raytrace(gp: &mut [Grid], par: &Parameters) -> Result<Option<RTPreparation>> {
1760    const NUM_FACES: usize = N_DIMS + 1;
1761    const N_FACES_INV: f64 = 1.0 / (NUM_FACES as f64);
1762    match par.ray_trace_algorithm {
1763        RayTraceAlgorithm::Modern => match delaunay(gp, par.ncell, true, false)? {
1764            DelaunayResult::Cells(mut cells) => {
1765                for (icell, cell) in cells.iter_mut().enumerate() {
1766                    for di in 0..N_DIMS {
1767                        let sum: f64 = (0..NUM_FACES)
1768                            .map(|vi| cell.vertex[vi].as_ref().map_or(0.0, |grid| grid.x[di]))
1769                            .sum();
1770                        cell.centre[di] = sum * N_FACES_INV;
1771                    }
1772                    cell.id = icell;
1773                }
1774
1775                let vertex_coords = extract_grid_xs(par.ncell, gp);
1776                let simplices = convert_cell_to_simplex(&cells, par, gp);
1777
1778                Ok(Some(RTPreparation {
1779                    simplices,
1780                    vertex_coords,
1781                }))
1782            }
1783            DelaunayResult::NoCells => Ok(None),
1784        },
1785        _ => Ok(None),
1786    }
1787}

1789pub fn raytrace(
1790    img: &mut Image,
1791    gp: &mut [Grid],
1792    par: &Parameters,
1793    mol_data: &[MolData],
1794    lam_kap: &Option<(RVector, RVector)>,
1795) -> Result<(), RTCError> {
1796    const MAX_NUM_RAYS_PER_PIXEL: usize = 20;
1797    const NUM_INTERP_PTS: usize = 3;
1798    const MIN_NUM_RAYS_FOR_AVERAGE: usize = 2;
1799
1800    let pixel_size = img.distance * img.img_res;
1801    let tot_n_img_pxls = (img.pxls * img.pxls) as usize;
1802    let img_centre_x_pxls = img.pxls as f64 / 2.0;
1803    let img_centre_y_pxls = img.pxls as f64 / 2.0;
1804
1805    if img.do_line {
1806        if img.trans > -1 {
1807            img.freq = mol_data[img.mol_i].freq[img.trans as usize];
1808        }
1809        if img.bandwidth > 0.0 && img.vel_res > 0.0 {
1810            img.nchan = (img.bandwidth / (img.vel_res / cc::SPEED_OF_LIGHT_SI * img.freq)) as usize;
1811        } else if img.bandwidth > 0.0 && img.nchan > 0 {
1812            img.vel_res = img.bandwidth * cc::SPEED_OF_LIGHT_SI / img.freq / img.nchan as f64;
1813        } else {
1814            img.bandwidth = img.nchan as f64 * img.vel_res / cc::SPEED_OF_LIGHT_SI * img.freq;
1815        }
1816    }
1817
1818    let cmb_freq = if img.do_line {
1819        let (cmb_mol_i, cmb_line_i) = if img.trans >= 0 {
1820            (img.mol_i, img.trans)
1821        } else {
1822            let mut min_freq = (img.freq - mol_data[0].freq[0]).abs();
1823            let mut cmb_mol_i = 0;
1824            let mut cmb_line_i = 0i64;
1825            for (mol_i, mol_data_i) in mol_data.iter().enumerate().take(par.n_species) {
1826                for line_i in 0..mol_data[mol_i].nline {
1827                    if mol_i == 0 && line_i == 0 {
1828                        continue;
1829                    }
1830                    let abs_delta_freq = img.freq - mol_data_i.freq[line_i];
1831                    if abs_delta_freq < min_freq {
1832                        min_freq = abs_delta_freq.abs();
1833                        cmb_mol_i = mol_i;
1834                        cmb_line_i = line_i as i64;
1835                    }
1836                }
1837            }
1838            (cmb_mol_i, cmb_line_i)
1839        };
1840        mol_data[cmb_mol_i].freq[cmb_line_i as usize]
1841    } else {
1842        img.freq
1843    };
1844
1845    let local_cmb = planck_fn(cmb_freq, cc::LOCAL_CMB_TEMP_SI);
1846    calc_grid_cont_dust_opacity(gp, par, cmb_freq, lam_kap)?;
1847
1848    for ppi in 0..tot_n_img_pxls {
1849        for ichan in 0..img.nchan {
1850            img.pixel[ppi].intense[ichan] = 0.0;
1851            img.pixel[ppi].tau[ichan] = 0.0;
1852        }
1853    }
1854
1855    for ppi in 0..tot_n_img_pxls {
1856        img.pixel[ppi].num_rays = 0;
1857    }
1858
1859    let mut num_pts_in_annulus = 0;
1860    for gpi in gp.iter().take(par.p_intensity) {
1861        let rsqu = gpi.x[0] * gpi.x[0] + gpi.x[1] * gpi.x[1];
1862        if rsqu > (4.0 / 9.0) * par.radius_squ {
1863            num_pts_in_annulus += 1;
1864        }
1865    }
1866    let num_circle_rays = if num_pts_in_annulus > 0 {
1867        let circle_spacing =
1868            (1.0 / 6.0) * par.radius * (5.0 * PI / num_pts_in_annulus as f64).sqrt();
1869        (2.0 * PI * par.radius / circle_spacing) as usize
1870    } else {
1871        0
1872    };
1873
1874    let mut xs = [0.0; 2];
1875
1876    let mut rays = Vec::with_capacity(par.p_intensity + num_circle_rays);
1877
1878    for gpi in gp.iter().take(par.p_intensity) {
1879        for (i, xsi) in xs.iter_mut().enumerate() {
1880            *xsi = (0..N_DIMS)
1881                .map(|d| gpi.x[d] * img.rotation_matrix[[d, i]])
1882                .sum();
1883        }
1884        if let Some(ray) = assign_ray_on_image(
1885            img,
1886            &xs,
1887            pixel_size,
1888            img_centre_x_pxls,
1889            img_centre_y_pxls,
1890            MAX_NUM_RAYS_PER_PIXEL,
1891        ) {
1892            rays.push(ray);
1893        }
1894    }
1895
1896    let num_active_rays_internal = rays.len();
1897    if num_circle_rays > 0 {
1898        let scale = 2.0 * PI / (num_circle_rays as f64);
1899        for i in 0..num_circle_rays {
1900            let angle = i as f64 * scale;
1901            xs[0] = par.radius * angle.cos();
1902            xs[1] = par.radius * angle.sin();
1903            if let Some(ray) = assign_ray_on_image(
1904                img,
1905                &xs,
1906                pixel_size,
1907                img_centre_x_pxls,
1908                img_centre_y_pxls,
1909                MAX_NUM_RAYS_PER_PIXEL,
1910            ) {
1911                rays.push(ray);
1912            }
1913        }
1914    }
1915
1916    let mut rtp_opt = prepare_raytrace(gp, par)?;
1917    let rtp = rtp_opt
1918        .as_mut()
1919        .expect("RTPreparation missing for Modern algorithm");
1920
1921    rays.par_iter_mut()
1922        .take(num_active_rays_internal)
1923        .try_for_each(|ray| {
1924            // Thread-local variables
1925            let mut gips = Vec::with_capacity(NUM_INTERP_PTS);
1926            for _ in 0..NUM_INTERP_PTS {
1927                gips.push(GridInterp::default());
1928            }
1929            match par.ray_trace_algorithm {
1930                RayTraceAlgorithm::Legacy => {
1931                    trace_ray(ray, gp, img, par, mol_data)?;
1932                }
1933                RayTraceAlgorithm::Modern => {
1934                    trace_ray_smooth(ray, &mut gips, gp, img, par, mol_data, rtp)?;
1935                }
1936            }
1937            Ok::<(), RTCError>(())
1938        })?;
1939
1940    for ray in rays.iter_mut().skip(num_active_rays_internal) {
1941        for ichan in 0..img.nchan {
1942            ray.intensity[ichan] = 0.0;
1943            ray.tau[ichan] = 0.0;
1944        }
1945    }
1946
1947    for ray in &mut rays {
1948        if ray.is_inside_image && img.pixel[ray.ppi].num_rays >= MIN_NUM_RAYS_FOR_AVERAGE {
1949            for ichan in 0..img.nchan {
1950                img.pixel[ray.ppi].intense[ichan] += ray.intensity[ichan];
1951                img.pixel[ray.ppi].tau[ichan] += ray.tau[ichan];
1952            }
1953        }
1954    }
1955
1956    let mut num_pixels_for_interp = 0;
1957    for ppi in 0..tot_n_img_pxls {
1958        if img.pixel[ppi].num_rays >= MIN_NUM_RAYS_FOR_AVERAGE {
1959            let nrays_inv = 1.0 / (img.pixel[ppi].num_rays as f64);
1960            for ichan in 0..img.nchan {
1961                img.pixel[ppi].intense[ichan] *= nrays_inv;
1962                img.pixel[ppi].tau[ichan] *= nrays_inv;
1963            }
1964        } else {
1965            num_pixels_for_interp += 1;
1966        }
1967    }
1968
1969    let mut raster_starts = RVector::zeros(2);
1970    let raster_dirs = RVector::from_vec(vec![0.0, 1.0]);
1971    let mut gis = UVector::zeros(3);
1972    let mut triangle = [[0.0; 2]; 3];
1973
1974    if num_pixels_for_interp > 0 {
1975        let mut raster_cell_ids = UVector::zeros(img.pxls as usize);
1976        let mut raster_pixel_is_in_cells: Vec<bool> = Vec::with_capacity(img.pxls as usize);
1977        let mut grid_2d_coords = RVector::zeros(2 * rays.len());
1978        for (ri, _) in rays.iter().enumerate() {
1979            grid_2d_coords[ri * 2] = rays[ri].x;
1980            grid_2d_coords[ri * 2 + 1] = rays[ri].y;
1981        }
1982        let simplices_2d = get_2d_cells(rays.as_slice(), rays.len())?;
1983        raster_starts[1] = pixel_size * (0.5 - img_centre_y_pxls);
1984        let rtp = RTPreparation {
1985            simplices: simplices_2d,
1986            vertex_coords: grid_2d_coords,
1987        };
1988        for xi in 0..img.pxls {
1989            let x = pixel_size * (0.5 + xi as f64 - img_centre_x_pxls);
1990            raster_starts[0] = x;
1991            for yi in 0..img.pxls {
1992                raster_pixel_is_in_cells[yi as usize] = false;
1993                raster_cell_ids[yi as usize] = 0;
1994            }
1995            let status = follow_ray_through_cells(2, &raster_starts, &raster_dirs, &rtp)?;
1996            let (entry_intersect_first_cell, chain_cell_ids, exit_intersects, len_chain_ptrs) =
1997                match status {
1998                    RTCResult::Success {
1999                        entry_intersect_first_cell,
2000                        chain_cell_ids,
2001                        exit_intersects,
2002                        len_chain_ptrs,
2003                    } => (
2004                        entry_intersect_first_cell,
2005                        chain_cell_ids,
2006                        exit_intersects,
2007                        len_chain_ptrs,
2008                    ),
2009                    _ => {
2010                        continue;
2011                    }
2012                };
2013            let mut start_yi = img.pxls; // default
2014            for yi in 0..img.pxls {
2015                let delta_y = pixel_size * yi as f64;
2016                if delta_y >= entry_intersect_first_cell.dist {
2017                    start_yi = yi;
2018                    break;
2019                }
2020            }
2021            let mut si = 0;
2022            for yi in start_yi..img.pxls {
2023                let delta_y = pixel_size * yi as f64;
2024                while si < len_chain_ptrs && delta_y >= exit_intersects[si].dist {
2025                    si += 1;
2026                }
2027                if si >= len_chain_ptrs {
2028                    break;
2029                }
2030                raster_cell_ids[yi as usize] = chain_cell_ids[si];
2031                raster_pixel_is_in_cells[yi as usize] = true;
2032            }
2033            for yi in 0..img.pxls {
2034                let ppi = yi * img.pxls + xi;
2035                if img.pixel[ppi as usize].num_rays >= MIN_NUM_RAYS_FOR_AVERAGE {
2036                    continue;
2037                }
2038                let y = pixel_size * (0.5 + yi as f64 - img_centre_y_pxls);
2039                if raster_pixel_is_in_cells[yi as usize] {
2040                    let dci = raster_cell_ids[yi as usize];
2041                    for vi in 0..3 {
2042                        gis[vi] = rtp.simplices[dci].vertex[vi];
2043                        triangle[vi][0] = rays[gis[vi]].x;
2044                        triangle[vi][1] = rays[gis[vi]].y;
2045                    }
2046                    let barys = calc_triangular_barycentric_coords(&triangle, x, y);
2047                    for ichan in 0..img.nchan {
2048                        let vals_intensity: RVector =
2049                            gis.iter().map(|&i| rays[i].intensity[ichan]).collect();
2050                        img.pixel[ppi as usize].intense[ichan] += vals_intensity.dot(&barys);
2051                        let vals_tau: RVector = gis.iter().map(|&i| rays[i].tau[ichan]).collect();
2052                        img.pixel[ppi as usize].tau[ichan] += vals_tau.dot(&barys);
2053                    }
2054                }
2055            }
2056        }
2057    }
2058
2059    let last_chan = if par.polarization { 0 } else { img.nchan };
2060
2061    for ppi in 0..tot_n_img_pxls {
2062        for ichan in 0..last_chan {
2063            img.pixel[ppi].intense[ichan] += ((-img.pixel[ppi].tau[ichan]).exp() - 1.0) * local_cmb;
2064        }
2065    }
2066
2067    Ok(())
2068}
