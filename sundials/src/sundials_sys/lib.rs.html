33#[cfg(test)]
34mod tests {
35    use crate::*;
36    use core::{ffi::c_void, ptr};
37
38    #[test]
39    #[allow(clippy::cast_possible_truncation)]
40    // This just tests if the most basic of all programs works. More tests to come soon.
41    fn simple_ode() {
42        unsafe extern "C" fn rhs(
43            _t: f64,
44            y: N_Vector,
45            dy: N_Vector,
46            _user_data: *mut c_void,
47        ) -> i32 {
48            unsafe {
49                *N_VGetArrayPointer(dy) = -*N_VGetArrayPointer(y);
50            }
51            0
52        }
53
54        unsafe {
55            let mut ctx = ptr::null_mut();
56            assert!(
57                SUNContext_Create(comm_no_mpi(), &raw mut ctx) >= 0,
58                "Could not initialize Context."
59            );
60            let y = N_VNew_Serial(1, ctx);
61            *N_VGetArrayPointer(y) = 1.0;
62
63            let mut cvode_mem = CVodeCreate(CV_ADAMS, ctx);
64
65            CVodeInit(cvode_mem, Some(rhs), 0.0, y);
66            CVodeSStolerances(cvode_mem, 1e-6, 1e-8);
67
68            let matrix = SUNDenseMatrix(1, 1, ctx);
69            let solver = SUNLinSol_Dense(y, matrix, ctx);
70
71            CVodeSetLinearSolver(cvode_mem, solver, matrix);
72
73            let mut t = 0f64;
74            CVode(cvode_mem, 1.0, y, &raw mut t, CV_NORMAL);
75            // y[0] is now exp(-1)
76
77            let result = (*N_VGetArrayPointer(y) * 1e6) as i32;
78            assert_eq!(result, 367_879);
79
80            N_VDestroy(y);
81            CVodeFree(&raw mut cvode_mem);
82            SUNLinSolFree(solver);
83            SUNMatDestroy(matrix);
84        }
85    }
