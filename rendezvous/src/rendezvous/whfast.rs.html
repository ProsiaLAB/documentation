99impl WHFast {
100    fn init(&mut self, ctx: &mut SyncContext<'_>) -> Result<(), WHFastError> {
101        if let Some(var_cfg) = ctx.var_cfg {
102            for v in var_cfg.iter() {
103                if v.order != 1 {
104                    return Err(WHFastError::UnsupportedVariationalOrder);
105                }
106                if v.is_test_particle {
107                    return Err(WHFastError::TestParticleVariationalNotSupported);
108                }
109            }
110        }
111
112        if ctx.var_cfg.is_some() && !matches!(self.coordinates, Coordinates::Jacobi) {
113            return Err(WHFastError::VariationalJacobiOnly);
114        }
115
116        if !matches!(self.kernel, Kernel::Default)
117            && !matches!(self.coordinates, Coordinates::Jacobi)
118        {
119            return Err(WHFastError::NonStandardKernelJacobiOnly);
120        }
121
122        if ctx.var_cfg.is_some() && !matches!(self.kernel, Kernel::Default) {
123            return Err(WHFastError::VariationalStandardKernelOnly);
124        }
125
126        if !matches!(self.corrector, Corrector::None)
127            && !matches!(
128                self.coordinates,
129                Coordinates::Jacobi | Coordinates::Barycentric
130            )
131        {
132            return Err(WHFastError::SymplecticCorrectorJacobiOrBarycentricOnly);
133        }
134
135        if self.keep_unsynchoronized && matches!(self.safe_mode, SafeMode::Combine) {
136            return Err(WHFastError::InvalidSafeModeCombination);
137        }
138
139        if matches!(self.kernel, Kernel::ModifiedKick | Kernel::Lazy) {
140            *ctx.gravity = Gravity::Jacobi;
141        } else {
142            match self.coordinates {
143                Coordinates::Jacobi => {
144                    *ctx.ignore_gravity_terms = IgnoreGravityTerms::IgnoreWHFastwithJacobi;
145                }
146                Coordinates::Barycentric => {
147                    *ctx.ignore_gravity_terms = IgnoreGravityTerms::IgnoreAll
148                }
149                _ => {
150                    *ctx.ignore_gravity_terms = IgnoreGravityTerms::IgnoreWHFastwithDHC;
151                }
152            }
153        }
154
155        self.recalculate_coordinates();
156
157        Ok(())
158    }

211impl Synchronize for WHFast {
212    fn synchronize(&mut self, ctx: &mut SyncContext<'_>) {
213        if self.init(ctx).is_err() {
214            return;
215        }
216
217        if !self.is_synchronized {
218            let n_real = ctx.particles.n_real();
219            let n_active = if ctx.particles.are_all_active()
220                || matches!(ctx.test_particle_kind, TestParticleKind::Massive)
221            {
222                n_real
223            } else {
224                ctx.particles.active.len()
225            };
226
227            let sync_particles = if self.keep_unsynchoronized {
228                Some(self.particles.clone())
229            } else {
230                None
231            };
232
233            match self.kernel {
234                Kernel::Default | Kernel::ModifiedKick => {}
235                Kernel::Lazy => {
236                    self.kepler_step(ctx, ctx.dt / 2.0);
237                    self.com_step(ctx.dt / 2.0);
238                }
239                Kernel::Composition => {
240                    self.kepler_step(ctx, 3.0 * ctx.dt / 8.0);
241                    self.com_step(3.0 * ctx.dt / 8.0);
242                }
243            }
244
245            if self.use_secondary_corrector {
246                self.apply_secondary_corrector();
247            }
248
249            if !matches!(self.corrector, Corrector::None) {
250                self.apply_corrector();
251            }
252
253            match self.coordinates {
254                Coordinates::Jacobi => {
255                    let masses = ctx
256                        .particles
257                        .active
258                        .iter()
259                        .map(|p| p.m)
260                        .collect::<Vec<f64>>();
261                    ctx.particles.active.transform_jacobi_to_inertial_posvel(
262                        &self.particles.active,
263                        &masses,
264                        n_real,
265                        n_active,
266                    );
267                }
268                Coordinates::DemocraticHeliocentric => {
269                    ctx.particles.active.transform_dhc_to_inertial_posvel(
270                        &self.particles.active,
271                        n_real,
272                        n_active,
273                    );
274                }
275                Coordinates::Whds => {
276                    ctx.particles.active.transform_whds_to_inertial_posvel(
277                        &self.particles.active,
278                        n_real,
279                        n_active,
280                    );
281                }
282                Coordinates::Barycentric => {
283                    ctx.particles
284                        .active
285                        .transform_barycentric_to_inertial_posvel(
286                            &self.particles.active,
287                            n_real,
288                            n_active,
289                        );
290                }
291            }
292
293            if let Some(vcs) = ctx.var_cfg {
294                todo!()
295            }
296
297            if let Some(sp) = sync_particles {
298                self.particles = sp;
299            }
300        }
301    }

388    pub(crate) fn gs3(gs: &mut [f64], beta: f64, x: f64) {
389        let x2 = x * x;
390        stumpff::cs3(gs, beta * x2);
391        gs[1] *= x;
392        gs[2] *= x2;
393        gs[3] *= x2 * x;

404    pub(crate) fn cs3(cs: &mut [f64], mut z: f64) {
405        let mut n = 0;
406
407        while z.abs() > 0.1 {
408            z /= 4.0;
409            n += 1;
410        }
411
412        const NMAX: usize = 13;
413
414        let mut c_odd = INVERSE_FACTORIALS[NMAX];
415        let mut c_even = INVERSE_FACTORIALS[NMAX - 1];
416
417        for np in (3..NMAX - 2).rev().step_by(2) {
418            c_odd = INVERSE_FACTORIALS[np] - z * c_odd;
419            c_even = INVERSE_FACTORIALS[np - 1] - z * c_even;
420        }
421
422        cs[3] = c_odd;
423        cs[2] = c_even;
424        cs[1] = INVERSE_FACTORIALS[1] - z * c_odd;
425        cs[0] = INVERSE_FACTORIALS[0] - z * c_even;
426
427        while n > 0 {
428            cs[3] = (cs[2] + cs[0] * cs[3]) * 0.25;
429            cs[2] = cs[1] * cs[1] * 0.5;
430            cs[1] *= cs[0];
431            cs[0] = 2.0 * cs[0] * cs[0] - 1.0;
432            n -= 1;
433        }
434    }

437struct KeplerSystem<'a> {
438    mass: f64,
439    r0: f64,
440    r0i: f64,
441    v2: f64,
442    beta: f64,
443    eta0: f64,
444    zeta0: f64,
445    gs: [f64; 6],
446    p: &'a mut Particle,
447}
448
449impl<'a> KeplerSystem<'a> {
450    fn new(p: &'a mut Particle, mass: f64) -> Self {
451        let r0 = (p.x * p.x + p.y * p.y + p.z * p.z).sqrt();
452        let r0i = 1.0 / r0;
453        let v2 = p.vx * p.vx + p.vy * p.vy + p.vz * p.vz;
454        let beta = 2.0 * mass * r0i - v2;
455        let eta0 = p.x * p.vx + p.y * p.vy + p.z * p.vz;
456        let zeta0 = mass - beta * r0;
457
458        KeplerSystem {
459            mass,
460            r0,
461            r0i,
462            v2,
463            beta,
464            eta0,
465            zeta0,
466            gs: [0.0; 6],
467            p,
468        }
469    }
470
471    fn solve(&mut self, dt: f64) {
472        struct BetaVals {
473            x_per_period: f64,
474            inv_period: f64,
475        }
476
477        let betvals = if self.beta > 0.0 {
478            Some(BetaVals {
479                x_per_period: 2.0 * PI * self.beta.sqrt(),
480                inv_period: self.beta.sqrt() * self.beta / (2.0 * PI * self.mass),
481            })
482        } else {
483            None
484        };
485
486        let mut nwarns = 0;
487
488        let mut x = if let Some(bv) = &betvals {
489            if dt.abs() * bv.inv_period > 1.0 && nwarns == 0 {
490                nwarns += 1;
491                eprintln!(
492                    "WARNING: WHFast is having convergence issues because the time step is \
493                     comparable to or larger than the orbital period (dt * inv_period = {}). \
494                     Consider reducing the time step.",
495                    dt.abs() * bv.inv_period
496                );
497            }
498
499            let dtr0i = dt * self.r0i;
500            dtr0i * (1.0 - dtr0i * self.eta0 * 0.5 * self.r0i)
501        } else {
502            0.0
503        };
504
505        let mut converged = false;
506        let mut old_x = x;
507
508        stiefel::gs3(&mut self.gs, self.beta, x);
509
510        let val = self.eta0 * self.gs[1] + self.zeta0 * self.gs[2];
511
512        let mut ri = 1.0 / (self.r0 + val);
513        x = ri * (x * val - self.eta0 * self.gs[2] - self.zeta0 * self.gs[3] + dt);
514
515        if let Some(bv) = &betvals {
516            if (x - old_x).abs() > 0.01 * bv.x_per_period {
517                // Quartic solver
518                // Linear initial guess
519                x = self.beta * dt / self.mass;
520                let mut prev_x = [0.0; NMAX_QUARTIC + 1];
521                for n_lag in 1..NMAX_QUARTIC {
522                    stiefel::gs3(&mut self.gs, self.beta, x);
523                    let f = self.r0 * x + self.eta0 * self.gs[2] + self.zeta0 * self.gs[3] - dt;
524                    let fp = self.r0 + self.eta0 * self.gs[1] + self.zeta0 * self.gs[2];
525                    let fpp = self.eta0 * self.gs[0] + self.zeta0 * self.gs[1];
526                    let denom = fp + (16.0 * fp * fp - 20.0 * f * fpp).abs().sqrt();
527                    let x_new = (x * denom - 5.0 * f) / denom;
528
529                    for &pxi in prev_x.iter().take(n_lag).skip(1) {
530                        if (x_new - pxi).abs() < 1e-12 {
531                            converged = true;
532                            break;
533                        }
534                    }
535
536                    if converged {
537                        break;
538                    }
539
540                    prev_x[n_lag] = x_new;
541                    x = x_new;
542                }
543                let val = self.eta0 * self.gs[1] + self.zeta0 * self.gs[2];
544                ri = 1.0 / (self.r0 + val);
545            } else {
546                // Newton's method
547
548                for _ in 1..NMAX_NEWTON {
549                    let old_x2 = old_x;
550                    old_x = x;
551                    stiefel::gs3(&mut self.gs, self.beta, x);
552                    let val = self.eta0 * self.gs[1] + self.zeta0 * self.gs[2];
553                    let ri = 1.0 / (self.r0 + val);
554                    x = ri * (x * val - self.eta0 * self.gs[2] - self.zeta0 * self.gs[3] + dt);
555                    if x == old_x || x == old_x2 {
556                        converged = true;
557                        break;
558                    }
559                }
560            }
561        }
562
563        if !converged {
564            let (mut xmin, mut xmax) = if let Some(bv) = &betvals {
565                let xmin = bv.x_per_period * (dt * bv.inv_period).floor();
566                let xmax = xmin + bv.x_per_period;
567                (xmin, xmax)
568            } else {
569                // Hyperbolic
570                let h2 = self.r0 * self.r0 * self.v2 - self.eta0 * self.eta0;
571                let q = h2
572                    / self.mass
573                    / (1.0 + (1.0 - h2 * self.beta / (self.mass * self.mass)).sqrt());
574                let vq = (h2.sqrt() / q).copysign(dt);
575                let mut xmin = dt / ((vq * dt).abs() + self.r0);
576                let mut xmax = dt / q;
577                if dt < 0.0 {
578                    mem::swap(&mut xmin, &mut xmax);
579                }
580                (xmin, xmax)
581            };
582            x = 0.5 * (xmin + xmax);
583
584            loop {
585                stiefel::gs3(&mut self.gs, self.beta, x);
586                let s = self.r0 * x + self.eta0 * self.gs[2] + self.zeta0 * self.gs[3] - dt;
587                if s >= 0.0 {
588                    xmax = x;
589                } else {
590                    xmin = x;
591                }
592                x = 0.5 * (xmin + xmax);
593
594                if (xmax - xmin).abs() > ((xmax + xmin) * 1e-15).abs() {
595                    break;
596                }
597            }
598            let val = self.eta0 * self.gs[1] + self.zeta0 * self.gs[2];
599            ri = 1.0 / (self.r0 + val);
600        }
601
602        if ri.is_nan() {
603            ri = 0.0;
604            self.gs[1] = 0.0;
605            self.gs[2] = 0.0;
606            self.gs[3] = 0.0;
607        }
608
609        let f = -self.mass * self.gs[2] * self.r0i;
610        let g = dt - self.mass * self.gs[3];
611        let fd = -self.mass * self.gs[1] * self.r0i * ri;
612        let gd = -self.mass * self.gs[2] * ri;
613    }
614}
