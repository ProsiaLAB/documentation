9const G: f64 = 2.9591220828559093e-4; // AU^3 / (day^2 * solar mass)

11pub struct Bodies {
12    pub q: Vec<Vec3>,
13    pub v: Vec<Vec3>,
14}
15
16impl Bodies {
17    pub fn momentum(&self, masses: &[f64]) -> Vec3 {
18        masses
19            .iter()
20            .zip(self.v.iter())
21            .map(|(&m, &vel)| vel * m)
22            .sum()
23    }
24
25    pub fn map_sun(&mut self, h: f64, masses: &[f64]) {
26        let mut sp = Vec3::zero();
27
28        for (&m, &v) in masses.iter().skip(1).zip(self.v.iter().skip(1)) {
29            sp += m * v;
30        }
31
32        let fac = h / masses[0] * sp;
33
34        for i in 1..masses.len() {
35            self.q[i] += fac;
36        }
37    }
38
39    pub fn adjust_sun(&mut self, _masses: &[f64], _p0: Vec3) {
40        // adjustSun
41    }
42}

44impl Bodies {
45    pub fn inter_pair(&mut self, h: f64, masses: &[f64], j: usize, k: usize) {
46        // Pair interaction
47        let q_vec = self.q[j] - self.q[k];
48        let q_mod = q_vec.norm();
49        let fac = G / q_mod.powi(3) * q_vec;
50        self.v[j] -= h * masses[k] * fac;
51        self.v[k] += h * masses[j] * fac;
52    }
53
54    pub fn kepler_pair(
55        &mut self,
56        h: f64,
57        masses: &[f64],
58        i: usize,
59        j: usize,
60    ) -> Result<(), IntegratorError> {
61        let gm = G * masses[0];
62        let a_i = &mut self.q[i];
63        let b_i = &mut self.v[i];
64
65        let mut kpl_i = KeplerSystem::new(a_i, b_i, gm, h);
66        kpl_i.step(0)?;
67
68        let a_j = &mut self.q[j];
69        let b_j = &mut self.v[j];
70        let mut kpl_j = KeplerSystem::new(a_j, b_j, gm, h);
71        kpl_j.step(0)?;
72
73        Ok(())
74    }
75}

84/// Symba: Symplectic Massive Body Algorithm
85pub struct Symba<'a> {
86    pub params: &'a SystemParams,
87    pub level_constraint: usize,
88}
89
90impl<'a> Symba<'a> {
91    pub fn step(&self, bodies: &mut Bodies) -> Result<(), IntegratorError> {
92        let n = self.params.masses.len();
93        let r_levels = self.params.r_levels.len();
94
95        let p0 = bodies.momentum(&self.params.masses);
96
97        bodies.map_sun(self.params.h_levels[0] / 2.0, &self.params.masses);
98
99        for level in 0..r_levels {
100            if self.level_constraint != level + 1 {
101                continue;
102            }
103
104            let stepsn = self.params.steps[level];
105            let hlevn = self.params.h_levels[level];
106
107            for j in 1..n {
108                for k in (j + 1)..n {
109                    for _ in 0..stepsn {
110                        bodies.inter_pair(hlevn / 2.0, &self.params.masses, j, k);
111                        bodies.kepler_pair(hlevn, &self.params.masses, j, k)?;
112                        bodies.inter_pair(hlevn / 2.0, &self.params.masses, j, k);
113                    }
114                }
115            }
116        }
117
118        bodies.map_sun(self.params.h_levels[0] / 2.0, &self.params.masses);
119
120        bodies.adjust_sun(&self.params.masses, p0);
121
122        Ok(())
123    }
124}

126struct KeplerSystem<'a> {
127    pub v: &'a mut Vec3,
128    pub vd: &'a mut Vec3,
129    pub kc: f64,
130    pub dt: f64,
131    pub r0: f64,
132    pub eta: f64,
133    pub beta: f64,
134    pub zeta: f64,
135    pub b: f64,
136}
137
138impl<'a> KeplerSystem<'a> {
139    fn new(v: &'a mut Vec3, vd: &'a mut Vec3, kc: f64, dt: f64) -> Self {
140        let v2 = vd.norm().powi(2);
141
142        let r0 = v.norm();
143        let eta = v.dot(vd);
144        let beta = 2.0 * kc / r0 - v2;
145        let zeta = kc - beta * r0;
146        let b = beta.abs().sqrt();
147        Self {
148            v,
149            vd,
150            kc,
151            dt,
152            r0,
153            eta,
154            beta,
155            zeta,
156            b,
157        }
158    }
159
160    fn initial_guess(&self) -> f64 {
161        if self.zeta != 0.0 {
162            let a = 3.0 * self.eta / self.zeta;
163            let b = 6.0 * self.r0 / self.zeta;
164            let c = -6.0 * self.dt / self.zeta;
165            cubic(a, b, c)
166        } else if self.eta != 0.0 {
167            let r_eta = self.r0 / self.eta;
168            let disc = r_eta * r_eta + 8.0 * self.dt / self.eta;
169            if disc >= 0.0 {
170                -r_eta + disc.sqrt()
171            } else {
172                self.dt / self.r0
173            }
174        } else {
175            self.dt / self.r0
176        }
177    }
178
179    fn step(&mut self, depth: u32) -> Result<(), IntegratorError> {
180        if depth > 30 {
181            panic!("Kepler step did not converge");
182        }
183
184        self.step_internal().or_else(|_| {
185            self.dt /= 4.0;
186            self.step(depth + 1)
187        })
188    }
189
190    fn step_internal(&mut self) -> Result<(), IntegratorError> {
191        let (g1, g2, r, ca, bsa) = if self.beta < 0.0 {
192            let x0 = self.initial_guess();
193            let mut x = x0;
194
195            let result = self.solve_universal_hyperbolic_newton(&mut x).or_else(|_| {
196                x = x0;
197                self.solve_universal_hyperbolic_laguerre(&mut x)
198            });
199
200            match result {
201                Ok((s2, c2)) => {
202                    let a = self.kc / (-self.beta);
203                    let g1 = 2.0 * s2 * c2 / self.b;
204                    let c = 2.0 * s2 * s2;
205                    let g2 = c / (-self.beta);
206                    let ca = c * a;
207                    let r = self.r0 + self.eta * g1 + self.zeta * g2;
208                    let bsa = (a / r) * (self.b / self.r0) * 2.0 * s2 * c2;
209                    (g1, g2, r, ca, bsa)
210                }
211                Err(_) => return Err(IntegratorError::KeplerStepFailed),
212            }
213        } else if self.beta > 0.0 {
214            let mut x0 = self.dt / self.r0;
215            let ff = self.zeta * x0 * x0 * x0 + 3.0 * self.eta * x0 * x0;
216            let fp = 3.0 * self.zeta * x0 * x0 + 6.0 * self.eta * x0 + 6.0 * self.r0;
217            x0 -= ff / fp;
218
219            let mut x = x0;
220            let result = self.solve_universal_hyperbolic_newton(&mut x).or_else(|_| {
221                x = x0;
222                self.solve_universal_hyperbolic_laguerre(&mut x)
223            });
224
225            match result {
226                Ok((s2, c2)) => {
227                    let a = self.kc / self.beta;
228                    let g1 = 2.0 * s2 * c2 / self.b;
229                    let c = 2.0 * s2 * s2;
230                    let g2 = c / self.beta;
231                    let ca = c * a;
232                    let r = self.r0 + self.eta * g1 + self.zeta * g2;
233                    let bsa = (a / r) * (self.b / self.r0) * 2.0 * s2 * c2;
234                    (g1, g2, r, ca, bsa)
235                }
236                Err(_) => return Err(IntegratorError::KeplerStepFailed),
237            }
238        } else {
239            let mut x = self.dt / self.r0;
240            let result = self.solve_universal_parabolic(&mut x);
241            match result {
242                Ok((_, _)) => {
243                    let g1 = x;
244                    let g2 = x * x / 2.0;
245                    let ca = self.kc * g2;
246                    let r = self.r0 + self.eta * g1 + self.zeta * g2;
247                    let bsa = (self.kc / x) / (r * self.r0);
248                    (g1, g2, r, ca, bsa)
249                }
250                Err(_) => return Err(IntegratorError::KeplerStepFailed),
251            }
252        };
253
254        let fhat = -(ca / self.r0);
255        let g = self.eta * g2 + self.r0 * g1;
256        let fdot = -bsa;
257        let gdothat = -(ca / r);
258
259        let s1 = self.v.x + fhat * self.v.x + g * self.vd.x;
260        let s2 = self.v.y + fhat * self.v.y + g * self.vd.y;
261        let s3 = self.v.z + fhat * self.v.z + g * self.vd.z;
262
263        let s4 = self.vd.x + fdot * self.v.x + gdothat * self.vd.x;
264        let s5 = self.vd.y + fdot * self.v.y + gdothat * self.vd.y;
265        let s6 = self.vd.z + fdot * self.v.z + gdothat * self.vd.z;
266
267        self.v.set(s1, s2, s3);
268        self.vd.set(s4, s5, s6);
269
270        Ok(())
271    }
272
273    fn solve_universal_parabolic(&self, x: &mut f64) -> Result<(f64, f64), ()> {
274        let a = 3.0 * self.eta / self.zeta;
275        let b = 6.0 * self.r0 / self.zeta;
276        let c = -6.0 * self.dt / self.zeta;
277        *x = cubic(a, b, c);
278        Ok((0.0, 1.0))
279    }
280
281    fn solve_universal_hyperbolic_newton(&self, x: &mut f64) -> Result<(f64, f64), ()> {
282        let mut xnew = *x;
283        let mut count = 0;
284        let mut flag = false;
285
286        while !flag {
287            let x0 = xnew;
288            let arg = self.b * x0 / 2.0;
289            if arg.abs() > 200.0 {
290                return Err(());
291            }
292            let s2 = arg.sinh();
293            let c2 = arg.cosh();
294            let g1 = 2.0 * s2 * c2 / self.b;
295            let g2 = 2.0 * s2 * s2 / (-self.beta);
296            let g3 = -(x0 - g1) / (-self.beta);
297            let g = self.eta * g1 + self.zeta * g2;
298            xnew = (x0 * g - self.eta * g2 - self.zeta * g3 + self.dt) / (self.r0 + g);
299            count += 1;
300            if count > 10 {
301                return Err(());
302            }
303            if (x0 - xnew) <= 1e-9 * xnew.abs() {
304                flag = true;
305            }
306        }
307
308        *x = xnew;
309        let arg = self.b * xnew / 2.0;
310        let s2 = arg.sinh();
311        let c2 = arg.cosh();
312        Ok((s2, c2))
313    }
314
315    fn solve_universal_hyperbolic_laguerre(&self, x: &mut f64) -> Result<(f64, f64), ()> {
316        let mut xnew = *x;
317        let mut count = 0;
318        let mut flag = false;
319
320        while !flag {
321            let x0 = xnew;
322            let arg = self.b * x0 / 2.0;
323            if arg.abs() > 50.0 {
324                return Err(());
325            }
326            let s2 = arg.sinh();
327            let c2 = arg.cosh();
328            let g1 = 2.0 * s2 * c2 / self.b;
329            let g2 = 2.0 * s2 * s2 / (-self.beta);
330            let g3 = -(x0 - g1) / (-self.beta);
331            let f = self.r0 * x0 + self.eta * g2 + self.zeta * g3 - self.dt;
332            let fp = self.r0 + self.eta * g1 + self.zeta * g2;
333            let g0 = 1.0 + (-self.beta * g2);
334            let fpp = self.eta * g0 + self.zeta * g1;
335            let denom = fp + (16.0 * fp * fp - 20.0 * f * fpp).abs().sqrt();
336            if denom == 0.0 {
337                return Err(());
338            }
339            let dx = -5.0 * f / denom;
340            xnew = x0 + dx;
341            count += 1;
342            if count > 20 {
343                return Err(());
344            }
345            if dx.abs() <= 1e-9 * xnew.abs() {
346                flag = true;
347            }
348        }
349        *x = xnew;
350        let arg = self.b * xnew / 2.0;
351        if arg.abs() > 200.0 {
352            return Err(());
353        }
354        let s2 = arg.sinh();
355        let c2 = arg.cosh();
356        let g1 = 2.0 * s2 * c2 / self.b;
357        let g2 = 2.0 * s2 * s2 / (-self.beta);
358        let g3 = -(*x - g1) / (-self.beta);
359        let g = self.eta * g1 + self.zeta * g2;
360        xnew = (*x * g - self.eta * g2 - self.zeta * g3 + self.dt) / (self.r0 + g);
361
362        *x = xnew;
363        let arg = self.b * xnew / 2.0;
364        let s2 = arg.sinh();
365        let c2 = arg.cosh();
366        Ok((s2, c2))
367    }

375fn cubic(a: f64, b: f64, c: f64) -> f64 {
376    let q = (a * a - 3.0 * b) / 9.0;
377    let r = (2.0 * a * a * a - 9.0 * a * b + 27.0 * c) / 54.0;
378
379    let r2 = r * r;
380    let q3 = q * q * q;
381
382    if r2 < q3 {
383        0.0
384    } else {
385        let sgn = r.signum();
386        let a = -sgn * (r.abs() + (r2 - q3).sqrt()).powf(1.0 / 3.0);
387        let b = if a == 0.0 { 0.0 } else { q / a };
388        (a + b) - a / 3.0
389    }
