11pub enum Boundary {
12    None,
13    Open,
14    Periodic,
15    Shear,
16}
17
18impl Boundary {
19    pub fn get_ghost_box(
20        &self,
21        ctx: &BoundaryContext<'_>,
22        i: isize,
23        j: isize,
24        k: isize,
25    ) -> GhostBox {
26        match self {
27            Boundary::Open => {
28                let x = ctx.box_size.x * (i as f64);
29                let y = ctx.box_size.y * (j as f64);
30                let z = ctx.box_size.z * (k as f64);
31                GhostBox {
32                    position: Vec3::new(x, y, z),
33                    velocity: Vec3::zero(),
34                }
35            }
36            Boundary::Shear => {
37                let omega = match ctx.integrator {
38                    Integrator::Sei(i) => i.omega,
39                    _ => {
40                        panic!("Shear boundary condition requires Sei integrator");
41                    }
42                };
43                let vx = 0.0;
44                let vy = -1.5 * (i as f64) * omega * ctx.box_size.x;
45                let vz = 0.0;
46
47                let shift = if i == 0 {
48                    -(vy * ctx.t) % ctx.box_size.y
49                } else if i > 0 {
50                    let rem = -(vy * ctx.t - ctx.box_size.y / 2.0) % ctx.box_size.y;
51                    rem - ctx.box_size.y / 2.0
52                } else {
53                    let rem = -(vy * ctx.t + ctx.box_size.y / 2.0) % ctx.box_size.y;
54                    rem + ctx.box_size.y / 2.0
55                };
56
57                let x = ctx.box_size.x * (i as f64);
58                let y = ctx.box_size.y * (j as f64) - shift;
59                let z = ctx.box_size.z * (k as f64);
60                GhostBox {
61                    position: Vec3::new(x, y, z),
62                    velocity: Vec3::new(vx, vy, vz),
63                }
64            }
65            Boundary::Periodic => {
66                let x = ctx.box_size.x * (i as f64);
67                let y = ctx.box_size.y * (j as f64);
68                let z = ctx.box_size.z * (k as f64);
69                GhostBox {
70                    position: Vec3::new(x, y, z),
71                    velocity: Vec3::zero(),
72                }
73            }
74            Boundary::None => GhostBox {
75                position: Vec3::zero(),
76                velocity: Vec3::zero(),
77            },
78        }
79    }
80}
