91/// Find a constant by its official name.
92///
93/// This searches both physical (CODATA) and astronomical (IAU) constants.
94///
95/// Returns `Some(Constant)` if found, `None` otherwise.
96///
97/// # Example
98/// ```
99/// use codata::find;
100/// let c = find("speed of light in vacuum").unwrap();
101/// assert_eq!(c.value, 299792458.0);
102///
103/// let au = find("Astronomical Unit").unwrap();
104/// assert_eq!(au.value, 149597870700.0);
105/// ```
106#[must_use]
107pub fn find(name: &str) -> Option<Constant> {
108    CONSTANTS
109        .iter()
110        .find(|(n, _)| n.eq_ignore_ascii_case(name))
111        .map(|(_, c)| *c)
112        .or_else(|| {
113            ASTRO_CONSTANTS
114                .iter()
115                .find(|(n, _)| n.eq_ignore_ascii_case(name))
116                .map(|(_, c)| *c)
117        })
118}

120#[cfg(test)]
121mod tests {

123
124    use super::*;
125    use crate::constants::SPEED_OF_LIGHT_IN_VACUUM;

127    #[test]
128    fn test_speed_of_light() {
129        assert_eq!(SPEED_OF_LIGHT_IN_VACUUM, 299_792_458.0);
130        assert_eq!(aliases::C, 299_792_458.0);
131    }
132
133    #[test]
134    fn test_find() {
135        let c = find("speed of light in vacuum").unwrap();
136        assert_eq!(c.value, 299_792_458.0);
137        assert_eq!(c.unit, "m s^-1");
138        assert_eq!(c.uncertainty, 0.0);
139
140        let au = find("Astronomical Unit").unwrap();
141        assert_eq!(au.value, 149_597_870_700.0);
142    }
143
144    #[test]
145    fn test_astro_aliases() {
146        assert_eq!(aliases::AU, 149_597_870_700.0);
147        const { assert!(aliases::M_SUN > 1.9e30) };
148    }
149
150    #[test]
151    fn test_gravitation() {
152        let g = find("Newtonian constant of gravitation").unwrap();
153        // CODATA 2022 value is 6.674 30(15) x 10^-11 m^3 kg^-1 s^-2
154        assert!((g.value - 6.674_30e-11).abs() < 1e-16);
155    }
156
157    #[test]
158    fn test_cgs_conversions() {
159        // G_cgs = G_si * 10^3
160        assert!((aliases::G_CGS / aliases::G_SI - 1000.0).abs() < 1e-10);
161        // c_cgs = c_si * 100
162        assert_eq!(aliases::C_CGS, aliases::C_SI * 100.0);
163        // h_cgs = h_si * 10^7 (J -> erg)
164        assert!((aliases::H_CGS / aliases::H_SI - 1e7).abs() < 1e-10);
165        // sigma_cgs = sigma_si * 10^3 (W/m^2/K^4 -> erg/s/cm^2/K^4)
166        // 10^7 (erg/s) / 10^4 (cm^2) = 10^3
167        assert!((aliases::SIGMA_CGS / aliases::SIGMA_SI - 1000.0).abs() < 1e-10);
168    }
169
170    #[test]
171    fn test_units() {
172        use crate::units::{length, time};
173        assert_eq!(time::MINUTE, 60.0);
174        assert_eq!(time::HOUR, 3600.0);
175        assert_eq!(length::INCH, 0.0254);
176        assert_eq!(length::LIGHT_YEAR, 9_460_730_472_580_800.0);
177    }
